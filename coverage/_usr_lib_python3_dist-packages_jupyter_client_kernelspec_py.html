39 def from_resource_dir(cls, resource_dir): 

40 """Create a KernelSpec object by reading kernel.json 

41 

42 Pass the path to the *directory* containing kernel.json. 

43 """ 

44 kernel_file = pjoin(resource_dir, 'kernel.json') 

45 with io.open(kernel_file, 'r', encoding='utf-8') as f: 

46 kernel_dict = json.load(f) 

47 return cls(resource_dir=resource_dir, **kernel_dict)

85def _list_kernels_in(dir): 

86 """Return a mapping of kernel names to resource directories from dir. 

87 

88 If dir is None or does not exist, returns an empty dict. 

89 """ 

90 if dir is None or not os.path.isdir(dir): 

91 return {} 

92 kernels = {} 

93 for f in os.listdir(dir): 

94 path = pjoin(dir, f) 

95 if not _is_kernel_dir(path): 

96 continue 

97 key = f.lower() 

98 if not _is_valid_kernel_name(key): 

99 warnings.warn("Invalid kernelspec directory name (%s): %s" 

100 % (_kernel_name_description, path), stacklevel=3, 

101 ) 

102 kernels[key] = path 

103 return kernels

161 def find_kernel_specs(self): 

162 """Returns a dict mapping kernel names to resource directories.""" 

163 d = {} 

164 for kernel_dir in self.kernel_dirs: 

165 kernels = _list_kernels_in(kernel_dir) 

166 for kname, spec in kernels.items(): 

167 if kname not in d: 

168 self.log.debug("Found kernel %s in %s", kname, kernel_dir) 

169 d[kname] = spec 

170 

171 if self.ensure_native_kernel and NATIVE_KERNEL_NAME not in d: 

172 try: 

173 from ipykernel.kernelspec import RESOURCES 

174 self.log.debug("Native kernel (%s) available from %s", 

175 NATIVE_KERNEL_NAME, RESOURCES) 

176 d[NATIVE_KERNEL_NAME] = RESOURCES 

177 except ImportError: 

178 self.log.warning("Native kernel (%s) is not available", NATIVE_KERNEL_NAME) 

179 

180 if self.whitelist: 

181 # filter if there's a whitelist 

182 d = {name:spec for name,spec in d.items() if name in self.whitelist} 

183 return d

202 def _find_spec_directory(self, kernel_name): 

203 """Find the resource directory of a named kernel spec""" 

204 for kernel_dir in self.kernel_dirs: 

205 try: 

206 files = os.listdir(kernel_dir) 

207 except OSError as e: 

208 if e.errno in (errno.ENOTDIR, errno.ENOENT): 

209 continue 

210 raise 

211 for f in files: 

212 path = pjoin(kernel_dir, f) 

213 if f.lower() == kernel_name and _is_kernel_dir(path): 

214 return path 

215 

216 if kernel_name == NATIVE_KERNEL_NAME: 

217 try: 

218 from ipykernel.kernelspec import RESOURCES 

219 except ImportError: 

220 pass 

221 else: 

222 return RESOURCES

224 def get_kernel_spec(self, kernel_name): 

225 """Returns a :class:`KernelSpec` instance for the given kernel_name. 

226 

227 Raises :exc:`NoSuchKernel` if the given kernel name is not found. 

228 """ 

229 if not _is_valid_kernel_name(kernel_name): 

230 self.log.warning("Kernelspec name %r is invalid: %s", kernel_name, 

231 _kernel_name_description) 

232 

233 resource_dir = self._find_spec_directory(kernel_name.lower()) 

234 if resource_dir is None: 

235 raise NoSuchKernel(kernel_name) 

236 

237 return self._get_kernel_spec_by_name(kernel_name, resource_dir)

239 def get_all_specs(self): 

240 """Returns a dict mapping kernel names to kernelspecs. 

241 

242 Returns a dict of the form:: 

243 

244 { 

245 'kernel_name': { 

246 'resource_dir': '/path/to/kernel_name', 

247 'spec': {"the spec itself": ...} 

248 }, 

249 ... 

250 } 

251 """ 

252 d = self.find_kernel_specs() 

253 res = {} 

254 for kname, resource_dir in d.items(): 

255 try: 

256 if self.__class__ is KernelSpecManager: 

257 spec = self._get_kernel_spec_by_name(kname, resource_dir) 

258 else: 

259 # avoid calling private methods in subclasses, 

260 # which may have overridden find_kernel_specs 

261 # and get_kernel_spec, but not the newer get_all_specs 

262 spec = self.get_kernel_spec(kname) 

263 

264 res[kname] = { 

265 "resource_dir": resource_dir, 

266 "spec": spec.to_dict() 

267 } 

268 except Exception: 

269 self.log.warning("Error loading kernelspec %r", kname, exc_info=True) 

270 return res

272 def remove_kernel_spec(self, name): 

273 """Remove a kernel spec directory by name. 

274 

275 Returns the path that was deleted. 

276 """ 

277 save_native = self.ensure_native_kernel 

278 try: 

279 self.ensure_native_kernel = False 

280 specs = self.find_kernel_specs() 

281 finally: 

282 self.ensure_native_kernel = save_native 

283 spec_dir = specs[name] 

284 self.log.debug("Removing %s", spec_dir) 

285 if os.path.islink(spec_dir): 

286 os.remove(spec_dir) 

287 else: 

288 shutil.rmtree(spec_dir) 

289 return spec_dir

300 def install_kernel_spec(self, source_dir, kernel_name=None, user=False, 

301 replace=None, prefix=None): 

302 """Install a kernel spec by copying its directory. 

303 

304 If ``kernel_name`` is not given, the basename of ``source_dir`` will 

305 be used. 

306 

307 If ``user`` is False, it will attempt to install into the systemwide 

308 kernel registry. If the process does not have appropriate permissions, 

309 an :exc:`OSError` will be raised. 

310 

311 If ``prefix`` is given, the kernelspec will be installed to 

312 PREFIX/share/jupyter/kernels/KERNEL_NAME. This can be sys.prefix 

313 for installation inside virtual or conda envs. 

314 """ 

315 source_dir = source_dir.rstrip('/\\') 

316 if not kernel_name: 

317 kernel_name = os.path.basename(source_dir) 

318 kernel_name = kernel_name.lower() 

319 if not _is_valid_kernel_name(kernel_name): 

320 raise ValueError("Invalid kernel name %r. %s" % (kernel_name, _kernel_name_description)) 

321 

322 if user and prefix: 

323 raise ValueError("Can't specify both user and prefix. Please choose one or the other.") 

324 

325 if replace is not None: 

326 warnings.warn( 

327 "replace is ignored. Installing a kernelspec always replaces an existing installation", 

328 DeprecationWarning, 

329 stacklevel=2, 

330 ) 

331 

332 destination = self._get_destination_dir(kernel_name, user=user, prefix=prefix) 

333 self.log.debug('Installing kernelspec in %s', destination) 

334 

335 kernel_dir = os.path.dirname(destination) 

336 if kernel_dir not in self.kernel_dirs: 

337 self.log.warning("Installing to %s, which is not in %s. The kernelspec may not be found.", 

338 kernel_dir, self.kernel_dirs, 

339 ) 

340 

341 if os.path.isdir(destination): 

342 self.log.info('Removing existing kernelspec in %s', destination) 

343 shutil.rmtree(destination) 

344 

345 shutil.copytree(source_dir, destination) 

346 self.log.info('Installed kernelspec %s in %s', kernel_name, destination) 

347 return destination

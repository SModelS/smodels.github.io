34def set_vml_accuracy_mode(mode): 

35 """ 

36 Set the accuracy mode for VML operations. 

37 

38 The `mode` parameter can take the values: 

39 - 'high': high accuracy mode (HA), <1 least significant bit 

40 - 'low': low accuracy mode (LA), typically 1-2 least significant bits 

41 - 'fast': enhanced performance mode (EP) 

42 - None: mode settings are ignored 

43 

44 This call is equivalent to the `vmlSetMode()` in the VML library. 

45 See: 

46 

47 http://www.intel.com/software/products/mkl/docs/webhelp/vml/vml_DataTypesAccuracyModes.html 

48 

49 for more info on the accuracy modes. 

50 

51 Returns old accuracy settings. 

52 """ 

53 if use_vml: 

54 acc_dict = {None: 0, 'low': 1, 'high': 2, 'fast': 3} 

55 acc_reverse_dict = {1: 'low', 2: 'high', 3: 'fast'} 

56 if mode not in list(acc_dict.keys()): 

57 raise ValueError( 

58 "mode argument must be one of: None, 'high', 'low', 'fast'") 

59 retval = _set_vml_accuracy_mode(acc_dict.get(mode, 0)) 

60 return acc_reverse_dict.get(retval) 

61 else: 

62 return None

81def set_num_threads(nthreads): 

82 """ 

83 Sets a number of threads to be used in operations. 

84 

85 Returns the previous setting for the number of threads. 

86 

87 During initialization time Numexpr sets this number to the number 

88 of detected cores in the system (see `detect_number_of_cores()`). 

89 

90 If you are using Intel's VML, you may want to use 

91 `set_vml_num_threads(nthreads)` to perform the parallel job with 

92 VML instead. However, you should get very similar performance 

93 with VML-optimized functions, and VML's parallelizer cannot deal 

94 with common expresions like `(x+1)*(x-2)`, while Numexpr's one 

95 can. 

96 """ 

97 old_nthreads = _set_num_threads(nthreads) 

98 return old_nthreads

100def _init_num_threads(): 

101 """ 

102 Detects the environment variable 'NUMEXPR_MAX_THREADS' to set the threadpool  

103 size, and if necessary the slightly redundant 'NUMEXPR_NUM_THREADS' or  

104 'OMP_NUM_THREADS' env vars to set the initial number of threads used by  

105 the virtual machine. 

106 """ 

107 # Any platform-specific short-circuits 

108 if 'sparc' in platform.machine(): 

109 log.warning('The number of threads have been set to 1 because problems related ' 

110 'to threading have been reported on some sparc machine. ' 

111 'The number of threads can be changed using the "set_num_threads" ' 

112 'function.') 

113 set_num_threads(1) 

114 return 1 

115 

116 env_configured = False 

117 n_cores = detect_number_of_cores() 

118 if 'NUMEXPR_MAX_THREADS' in os.environ: 

119 # The user has configured NumExpr in the expected way, so suppress logs. 

120 env_configured = True 

121 n_cores = MAX_THREADS 

122 else: 

123 # The use has not set 'NUMEXPR_MAX_THREADS', so likely they have not  

124 # configured NumExpr as desired, so we emit info logs. 

125 if n_cores > MAX_THREADS: 

126 log.info('Note: detected %d virtual cores but NumExpr set to maximum of %d, check "NUMEXPR_MAX_THREADS" environment variable.'%(n_cores, MAX_THREADS)) 

127 if n_cores > 8: 

128 # The historical 'safety' limit. 

129 log.info('Note: NumExpr detected %d cores but "NUMEXPR_MAX_THREADS" not set, so enforcing safe limit of 8.'%n_cores) 

130 n_cores = 8 

131 

132 # Now we check for 'NUMEXPR_NUM_THREADS' or 'OMP_NUM_THREADS' to set the  

133 # actual number of threads used. 

134 if 'NUMEXPR_NUM_THREADS' in os.environ: 

135 requested_threads = int(os.environ['NUMEXPR_NUM_THREADS']) 

136 elif 'OMP_NUM_THREADS' in os.environ: 

137 requested_threads = int(os.environ['OMP_NUM_THREADS']) 

138 else: 

139 requested_threads = n_cores 

140 if not env_configured: 

141 log.info('NumExpr defaulting to %d threads.'%n_cores) 

142 

143 # The C-extension function performs its own checks against `MAX_THREADS` 

144 set_num_threads(requested_threads) 

145 return requested_threads

148def detect_number_of_cores(): 

149 """ 

150 Detects the number of cores on a system. Cribbed from pp. 

151 """ 

152 # Linux, Unix and MacOS: 

153 if hasattr(os, "sysconf"): 

154 if "SC_NPROCESSORS_ONLN" in os.sysconf_names: 

155 # Linux & Unix: 

156 ncpus = os.sysconf("SC_NPROCESSORS_ONLN") 

157 if isinstance(ncpus, int) and ncpus > 0: 

158 return ncpus 

159 else: # OSX: 

160 return int(subprocess.check_output(["sysctl", "-n", "hw.ncpu"])) 

161 # Windows: 

162 try: 

163 ncpus = int(os.environ.get("NUMBER_OF_PROCESSORS", "")) 

164 if ncpus > 0: 

165 return ncpus 

166 except ValueError: 

167 pass 

168 return 1 # Default

191class CacheDict(dict): 

192 """ 

193 A dictionary that prevents itself from growing too much. 

194 """ 

195 

196 def __init__(self, maxentries): 

197 self.maxentries = maxentries 

198 super(CacheDict, self).__init__(self) 

199 

200 def __setitem__(self, key, value): 

201 # Protection against growing the cache too much 

202 if len(self) > self.maxentries: 

203 # Remove a 10% of (arbitrary) elements from the cache 

204 entries_to_remove = self.maxentries // 10 

205 for k in list(self.keys())[:entries_to_remove]: 

206 super(CacheDict, self).__delitem__(k) 

207 super(CacheDict, self).__setitem__(key, value)

24def indent(instr,nspaces=4, ntabs=0, flatten=False): 

25 """Indent a string a given number of spaces or tabstops. 

26 

27 indent(str,nspaces=4,ntabs=0) -> indent str by ntabs+nspaces. 

28 

29 Parameters 

30 ---------- 

31 

32 instr : basestring 

33 The string to be indented. 

34 nspaces : int (default: 4) 

35 The number of spaces to be indented. 

36 ntabs : int (default: 0) 

37 The number of tabs to be indented. 

38 flatten : bool (default: False) 

39 Whether to scrub existing indentation. If True, all lines will be 

40 aligned to the same indentation. If False, existing indentation will 

41 be strictly increased. 

42 

43 Returns 

44 ------- 

45 

46 str|unicode : string indented by ntabs and nspaces. 

47 

48 """ 

49 if instr is None: 

50 return 

51 ind = '\t'*ntabs+' '*nspaces 

52 if flatten: 

53 pat = re.compile(r'^\s*', re.MULTILINE) 

54 else: 

55 pat = re.compile(r'^', re.MULTILINE) 

56 outstr = re.sub(pat, ind, instr) 

57 if outstr.endswith(os.linesep+ind): 

58 return outstr[:-len(ind)] 

59 else: 

60 return outstr

63def dedent(text): 

64 """Equivalent of textwrap.dedent that ignores unindented first line. 

65 

66 This means it will still dedent strings like: 

67 '''foo 

68 is a bar 

69 ''' 

70 

71 For use in wrap_paragraphs. 

72 """ 

73 

74 if text.startswith('\n'): 

75 # text starts with blank line, don't ignore the first line 

76 return textwrap.dedent(text) 

77 

78 # split first line 

79 splits = text.split('\n',1) 

80 if len(splits) == 1: 

81 # only one line 

82 return textwrap.dedent(text) 

83 

84 first, rest = splits 

85 # dedent everything but the first line 

86 rest = textwrap.dedent(rest) 

87 return '\n'.join([first, rest])

90def wrap_paragraphs(text, ncols=80): 

91 """Wrap multiple paragraphs to fit a specified width. 

92 

93 This is equivalent to textwrap.wrap, but with support for multiple 

94 paragraphs, as separated by empty lines. 

95 

96 Returns 

97 ------- 

98 

99 list of complete paragraphs, wrapped to fill `ncols` columns. 

100 """ 

101 paragraph_re = re.compile(r'\n(\s*\n)+', re.MULTILINE) 

102 text = dedent(text).strip() 

103 paragraphs = paragraph_re.split(text)[::2] # every other entry is space 

104 out_ps = [] 

105 indent_re = re.compile(r'\n\s+', re.MULTILINE) 

106 for p in paragraphs: 

107 # presume indentation that survives dedent is meaningful formatting, 

108 # so don't fill unless text is flush. 

109 if indent_re.search(p) is None: 

110 # wrap paragraph 

111 p = textwrap.fill(p, ncols) 

112 out_ps.append(p) 

113 return out_ps

132def _chunks(l, n): 

133 """Yield successive n-sized chunks from l.""" 

134 for i in range(0, len(l), n): 

135 yield l[i:i+n]

138def _find_optimal(rlist , separator_size=2 , displaywidth=80): 

139 """Calculate optimal info to columnize a list of string""" 

140 for nrow in range(1, len(rlist)+1) : 

141 chk = list(map(max,_chunks(rlist, nrow))) 

142 sumlength = sum(chk) 

143 ncols = len(chk) 

144 if sumlength+separator_size*(ncols-1) <= displaywidth : 

145 break; 

146 return {'columns_numbers' : ncols, 

147 'optimal_separator_width':(displaywidth - sumlength)/(ncols-1) if (ncols -1) else 0, 

148 'rows_numbers' : nrow, 

149 'columns_width' : chk 

150 }

161def compute_item_matrix(items, empty=None, *args, **kwargs) : 

162 """Returns a nested list, and info to columnize items 

163 

164 Parameters 

165 ---------- 

166 

167 items 

168 list of strings to columize 

169 empty : (default None) 

170 default value to fill list if needed 

171 separator_size : int (default=2) 

172 How much caracters will be used as a separation between each columns. 

173 displaywidth : int (default=80) 

174 The width of the area onto wich the columns should enter 

175 

176 Returns 

177 ------- 

178 

179 strings_matrix 

180 

181 nested list of string, the outer most list contains as many list as 

182 rows, the innermost lists have each as many element as colums. If the 

183 total number of elements in `items` does not equal the product of 

184 rows*columns, the last element of some lists are filled with `None`. 

185 

186 dict_info 

187 some info to make columnize easier: 

188 

189 columns_numbers 

190 number of columns 

191 rows_numbers 

192 number of rows 

193 columns_width 

194 list of with of each columns 

195 optimal_separator_width 

196 best separator width between columns 

197 

198 Examples 

199 -------- 

200 :: 

201 

202 In [1]: l = ['aaa','b','cc','d','eeeee','f','g','h','i','j','k','l'] 

203 ...: compute_item_matrix(l,displaywidth=12) 

204 Out[1]: 

205 ([['aaa', 'f', 'k'], 

206 ['b', 'g', 'l'], 

207 ['cc', 'h', None], 

208 ['d', 'i', None], 

209 ['eeeee', 'j', None]], 

210 {'columns_numbers': 3, 

211 'columns_width': [5, 1, 1], 

212 'optimal_separator_width': 2, 

213 'rows_numbers': 5}) 

214 """ 

215 info = _find_optimal(list(map(len, items)), *args, **kwargs) 

216 nrow, ncol = info['rows_numbers'], info['columns_numbers'] 

217 return ([[ _get_or_default(items, c*nrow+i, default=empty) for c in range(ncol) ] for i in range(nrow) ], info)

220def columnize(items, separator=' ', displaywidth=80): 

221 """ Transform a list of strings into a single string with columns. 

222 

223 Parameters 

224 ---------- 

225 items : sequence of strings 

226 The strings to process. 

227 

228 separator : str, optional [default is two spaces] 

229 The string that separates columns. 

230 

231 displaywidth : int, optional [default is 80] 

232 Width of the display in number of characters. 

233 

234 Returns 

235 ------- 

236 The formatted string. 

237 """ 

238 if not items : 

239 return '\n' 

240 matrix, info = compute_item_matrix(items, separator_size=len(separator), displaywidth=displaywidth) 

241 fmatrix = [filter(None, x) for x in matrix] 

242 sjoin = lambda x : separator.join([ y.ljust(w, ' ') for y, w in zip(x, info['columns_width'])]) 

243 return '\n'.join(map(sjoin, fmatrix))+'\n'

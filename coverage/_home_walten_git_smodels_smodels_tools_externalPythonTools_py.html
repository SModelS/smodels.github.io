14class ExternalPythonTool(object): 

15 """ 

16 An instance of this class represents the installation of a python package. 

17 As it is python-only, we need this only for installation, not for running 

18 (contrary to nllfast or pythia).  

19  

20 """ 

21 def __init__(self, importname, optional=False ): 

22 """ 

23 Initializes the ExternalPythonTool object. Useful for installation.  

24 :params optional: optional package, not needed for core SModelS. 

25 """ 

26 self.name = importname 

27 self.optional = optional 

28 self.pythonPath = "" 

29 try: 

30 i = __import__(importname) 

31 self.pythonPath = i.__file__.replace("/__init__.pyc", "") 

32 except ImportError as e: 

33 if optional: 

34 logger.debug("could not find %s: %s (but its not necessary for smodels, so dont worry)" % (importname, e)) 

35 else: 

36 logger.error("could not find %s: %s" % (importname, e)) 

37 

38 def compile ( self ): 

39 try: 

40 import pip 

41 pip.main(["install","--user",self.name] ) 

42 return 

43 except (ImportError,AttributeError): 

44 pass 

45 try: 

46 import pip._internal 

47 pip._internal.main(["install","--user",self.name] ) 

48 return 

49 except (ImportError,AttributeError): 

50 pass 

51 try: 

52 from setuptools.command import easy_install 

53 easy_install.main(["-U","--user",self.name]) 

54 return 

55 except (ImportError,AttributeError): 

56 pass

24 @key_bindings.add(Keys.Vt100MouseEvent) 

25 def _(event): 

26 """ 

27 Handling of incoming mouse event. 

28 """ 

29 # TypicaL: "eSC[MaB*" 

30 # Urxvt: "Esc[96;14;13M" 

31 # Xterm SGR: "Esc[<64;85;12M" 

32 

33 # Parse incoming packet. 

34 if event.data[2] == 'M': 

35 # Typical. 

36 mouse_event, x, y = map(ord, event.data[3:]) 

37 mouse_event = { 

38 32: MouseEventType.MOUSE_DOWN, 

39 35: MouseEventType.MOUSE_UP, 

40 96: MouseEventType.SCROLL_UP, 

41 97: MouseEventType.SCROLL_DOWN, 

42 }.get(mouse_event) 

43 

44 # Handle situations where `PosixStdinReader` used surrogateescapes. 

45 if x >= 0xdc00: x -= 0xdc00 

46 if y >= 0xdc00: y -= 0xdc00 

47 

48 x -= 32 

49 y -= 32 

50 else: 

51 # Urxvt and Xterm SGR. 

52 # When the '<' is not present, we are not using the Xterm SGR mode, 

53 # but Urxvt instead. 

54 data = event.data[2:] 

55 if data[:1] == '<': 

56 sgr = True 

57 data = data[1:] 

58 else: 

59 sgr = False 

60 

61 # Extract coordinates. 

62 mouse_event, x, y = map(int, data[:-1].split(';')) 

63 m = data[-1] 

64 

65 # Parse event type. 

66 if sgr: 

67 mouse_event = { 

68 (0, 'M'): MouseEventType.MOUSE_DOWN, 

69 (0, 'm'): MouseEventType.MOUSE_UP, 

70 (64, 'M'): MouseEventType.SCROLL_UP, 

71 (65, 'M'): MouseEventType.SCROLL_DOWN, 

72 }.get((mouse_event, m)) 

73 else: 

74 mouse_event = { 

75 32: MouseEventType.MOUSE_DOWN, 

76 35: MouseEventType.MOUSE_UP, 

77 96: MouseEventType.SCROLL_UP, 

78 97: MouseEventType.SCROLL_DOWN, 

79 }.get(mouse_event) 

80 

81 x -= 1 

82 y -= 1 

83 

84 # Only handle mouse events when we know the window height. 

85 if event.app.renderer.height_is_known and mouse_event is not None: 

86 # Take region above the layout into account. The reported 

87 # coordinates are absolute to the visible part of the terminal. 

88 try: 

89 y -= event.app.renderer.rows_above_layout 

90 except HeightIsUnknownError: 

91 return 

92 

93 # Call the mouse handler from the renderer. 

94 handler = event.app.renderer.mouse_handlers.mouse_handlers[x, y] 

95 handler(MouseEvent(position=Point(x=x, y=y), 

96 event_type=mouse_event))

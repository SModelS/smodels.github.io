27class Waker(interface.Waker): 

28 """Create an OS independent asynchronous pipe. 

29 

30 For use on platforms that don't have os.pipe() (or where pipes cannot 

31 be passed to select()), but do have sockets. This includes Windows 

32 and Jython. 

33 """ 

34 def __init__(self): 

35 from .auto import set_close_exec 

36 # Based on Zope select_trigger.py: 

37 # https://github.com/zopefoundation/Zope/blob/master/src/ZServer/medusa/thread/select_trigger.py 

38 

39 self.writer = socket.socket() 

40 set_close_exec(self.writer.fileno()) 

41 # Disable buffering -- pulling the trigger sends 1 byte, 

42 # and we want that sent immediately, to wake up ASAP. 

43 self.writer.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1) 

44 

45 count = 0 

46 while 1: 

47 count += 1 

48 # Bind to a local port; for efficiency, let the OS pick 

49 # a free port for us. 

50 # Unfortunately, stress tests showed that we may not 

51 # be able to connect to that port ("Address already in 

52 # use") despite that the OS picked it. This appears 

53 # to be a race bug in the Windows socket implementation. 

54 # So we loop until a connect() succeeds (almost always 

55 # on the first try). See the long thread at 

56 # http://mail.zope.org/pipermail/zope/2005-July/160433.html 

57 # for hideous details. 

58 a = socket.socket() 

59 set_close_exec(a.fileno()) 

60 a.bind(("127.0.0.1", 0)) 

61 a.listen(1) 

62 connect_address = a.getsockname() # assigned (host, port) pair 

63 try: 

64 self.writer.connect(connect_address) 

65 break # success 

66 except socket.error as detail: 

67 if (not hasattr(errno, 'WSAEADDRINUSE') or 

68 errno_from_exception(detail) != errno.WSAEADDRINUSE): 

69 # "Address already in use" is the only error 

70 # I've seen on two WinXP Pro SP2 boxes, under 

71 # Pythons 2.3.5 and 2.4.1. 

72 raise 

73 # (10048, 'Address already in use') 

74 # assert count <= 2 # never triggered in Tim's tests 

75 if count >= 10: # I've never seen it go above 2 

76 a.close() 

77 self.writer.close() 

78 raise socket.error("Cannot bind trigger!") 

79 # Close `a` and try again. Note: I originally put a short 

80 # sleep() here, but it didn't appear to help or hurt. 

81 a.close() 

82 

83 self.reader, addr = a.accept() 

84 set_close_exec(self.reader.fileno()) 

85 self.reader.setblocking(0) 

86 self.writer.setblocking(0) 

87 a.close() 

88 self.reader_fd = self.reader.fileno()

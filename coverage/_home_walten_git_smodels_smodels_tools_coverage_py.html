79 def __init__(self,topoList, sqrts=None, sigmacut=0*fb, 

80 groupFilters = filtersDefault, 

81 groupFactors = factorsDefault, 

82 groupdDescriptions = descriptionDefault, 

83 smFinalStates=None,bsmFinalSates=None): 

84 """ 

85 Inititalize the object. 

86 

87 :ivar topoList: TopologyList object used to select elements from. 

88 :ivar sqrts: Value (with units) for the center of mass energy used to compute the missing cross sections. 

89 If not specified the largest value available will be used. 

90 :ivar sigmacut: Minimum cross-section/weight value (after applying the reweight factor) 

91 for an element to be included. The value should in fb (unitless) 

92 :ivar groupFilters: Dictionary containing the groups' labels and the method for selecting 

93 elements. 

94 :ivar groupFactors: Dictionary containing the groups' labels and the method for reweighting 

95 cross sections. 

96 :ivar groupdDescriptions: Dictionary containing the groups' labels and strings describing the group 

97 (used for printout) 

98 :ivar smFinalStates: List of (inclusive) Particle or MultiParticle objects used for grouping Z2-even 

99 particles when creating GeneralElements. 

100 :ivar bsmFinalSates: List of (inclusive) Particle or MultiParticle objects used for grouping Z2-odd 

101 particles when creating GeneralElements. 

102 """ 

103 #Sanity checks: 

104 if not isinstance(groupFilters,dict): 

105 raise SModelSError("groupFilters input should be a Dictionary and not %s" %type(groupFilters)) 

106 if not isinstance(groupFactors,dict): 

107 raise SModelSError("groupFactors input should be a Dictionary and not %s" %type(groupFactors)) 

108 if sorted(groupFilters.keys()) != sorted(groupFactors.keys()): 

109 raise SModelSError("Keys in groupFilters and groupFactors do not match") 

110 if any(not hasattr(gFilter,'__call__') for gFilter in groupFilters.values()): 

111 raise SModelSError("Group filters must be callable methods") 

112 

113 

114 if smFinalStates is None: 

115 #Define multiparticles for conveniently grouping SM final states 

116 taList = MultiParticle('ta', [ta,taC]) 

117 lList = MultiParticle('l', [e,mu,eC,muC]) 

118 WList = MultiParticle('W', [W,WC]) 

119 tList = MultiParticle('t', [t,tC]) 

120 jetList = MultiParticle('jet', [q,c,g,pion]) 

121 nuList = nu 

122 smFinalStates = [WList, lList, tList, taList, nuList, jetList] 

123 if bsmFinalSates is None: 

124 #Define inclusive BSM states to group/label the last BSM states: 

125 bsmFinalStates = bsmDefault 

126 

127 if sqrts is None: 

128 sqrts = max([xsec.info.sqrts for xsec in topoList.getTotalWeight()]) 

129 else: 

130 sqrts = sqrts 

131 

132 #Store the relevant element cross-sections to improve performance: 

133 for el in topoList.getElements(): 

134 xsec = el.weight.getXsecsFor(sqrts) 

135 if xsec: 

136 el._totalXsec = xsec[0].value.asNumber(fb) 

137 else: 

138 el._totalXsec = 0. 

139 

140 self.groups = [] 

141 #Create each uncovered group and get the topologies from topoList 

142 for gLabel,gFilter in groupFilters.items(): 

143 #Initialize the uncovered topology list: 

144 uncoveredTopos = UncoveredGroup(label=gLabel,elementFilter=gFilter, 

145 reweightFactor = groupFactors[gLabel], 

146 smFinalStates=smFinalStates, 

147 bsmFinalStates=bsmFinalStates, 

148 sqrts=sqrts, sigmacut=sigmacut.asNumber(fb)) 

149 if groupdDescriptions and gLabel in groupdDescriptions: 

150 uncoveredTopos.description = groupdDescriptions[gLabel] 

151 else: 

152 uncoveredTopos.description = gLabel 

153 #Fill the list with the elements in topoList: 

154 uncoveredTopos.getToposFrom(topoList) 

155 self.groups.append(uncoveredTopos)

172class UncoveredGroup(object): 

173 """ 

174 Holds information about a single coverage group: criteria for selecting and grouping elements, 

175 function for reweighting cross sections, etc. 

176 

177 :ivar label: Group label 

178 :ivar elementFilter: Function which takes an element as argument and returns True (False) if 

179 the element should (not) be selected. 

180 :ivar reweightFactor: Function which takes an element as argument and returns the reweighting 

181 factor to be applied to the element weight. 

182 :ivar smFinalStates: List of Particle/MultiParticle objects used to group Z2-even particles appearing 

183 in the final state 

184 :ivar bsmFinalStates: List of Particle/MultiParticle objects used to group Z2-odd particles appearing 

185 in the final state 

186 :ivar sqrts: Value (with units) for the center of mass energy used to compute the missing cross sections. 

187 If not specified the largest value available will be used. 

188 :ivar sigmacut: Minimum cross-section/weight value (after applying the reweight factor) 

189 for an element to be included. The value should in fb (unitless) 

190 """ 

191 def __init__(self, label, elementFilter, reweightFactor, 

192 smFinalStates, bsmFinalStates, sqrts, sigmacut=0.): 

193 self.generalElements = [] 

194 self.smFinalStates = smFinalStates 

195 self.bsmFinalStates = bsmFinalStates 

196 self.sqrts = sqrts 

197 self.sigmacut = sigmacut 

198 self.label = label 

199 self.elementFilter = elementFilter 

200 self.reweightFactor = reweightFactor

208 def getToposFrom(self,topoList): 

209 """ 

210 Select the elements from topoList according to self.elementFilter 

211 and build GeneralElements from the selected elements. 

212 The GeneralElement weights corresponds to the missing cross-section 

213 with double counting from compressed elements already accounted for. 

214 """ 

215 

216 #First select all elements according to the filter (type of uncovered/missing topology): 

217 elementList = [el for el in topoList.getElements() if self.elementFilter(el)] 

218 

219 #Get missing xsections including the reweight factor: 

220 missingXandEls = [[self.getMissingX(el)*self.reweightFactor(el),el] for el in elementList] 

221 #Only keep the ones elements sigmacut: 

222 if self.sigmacut: 

223 missingXandEls = [x for x in missingXandEls[:] if x[0] > self.sigmacut] 

224 else: 

225 missingXandEls = [x for x in missingXandEls[:] if x[0] > 0.] 

226 #Sort according to largest missingX, smallest size and largest ID 

227 missingXandEls = sorted(missingXandEls, key = lambda pt: [pt[0],-pt[1]._getLength(),-pt[1].elID], reverse=True) 

228 

229 #Split lists of elements and missingX: 

230 missingXsecs = [pt[0] for pt in missingXandEls] 

231 elementList = [pt[1] for pt in missingXandEls] 

232 

233 #Remove all elements which are related to each other in order to avoid double counting 

234 #(keep always the first appearance in the list, so we always keep the ones with largest missing xsec) 

235 elementListUnique = [] 

236 missingXsecsUnique = [] 

237 ancestors = set() #Keep track of all the ancestor ids of the elements in the unique list 

238 for i,element in enumerate(elementList): 

239 #Get ancestor object ids 

240 #(safer than element.elID since some elements can share elID = 0 if they 

241 #were never directly inserted in the TopologyList) 

242 ancestorsIDs = set([id(element)] + [id(el) for el in element.getAncestors()]) 

243 #If the element has any common ancestor with any of the previous elements 

244 #or if it is an ancestor of any of the previous elements 

245 #skip it to avoid double counting 

246 if ancestors.intersection(ancestorsIDs): 

247 continue 

248 elementListUnique.append(element) 

249 missingXsecsUnique.append(missingXsecs[i]) 

250 ancestors = ancestors.union(ancestorsIDs) 

251 

252 #Now that we only have unique elements with their effective missing cross-sections 

253 #we create General Elements out of them 

254 for i,el in enumerate(elementListUnique): 

255 missingX = missingXsecsUnique[i] 

256 if not missingX: 

257 continue 

258 self.addToGeneralElements(el,missingX) 

259 

260 #Finally sort general elements by their missing cross-section: 

261 self.generalElements = sorted(self.generalElements[:], key = lambda genEl: genEl.missingX, reverse=True)

263 def getMissingX(self,element): 

264 """ 

265 Calculate total missing cross section of an element, by recursively checking if its 

266 mothers already appear in the list. 

267 :param element: Element object 

268 

269 :returns: missing cross section without units (in fb) 

270 """ 

271 

272 ancestorList = element.getAncestors() 

273 alreadyChecked = [] # keep track of which elements have already been checked 

274 #Get the (pre-loaded) total element weight in fb: 

275 missingX = element._totalXsec 

276 if not missingX: 

277 return 0. 

278 overlapXsec = 0. 

279 for ancestor in ancestorList: #check all ancestors (ancestorList is sorted by generation) 

280 #Skip entries which correspond to the element itself 

281 if ancestor is element: 

282 continue 

283 #Make sure we do not subtract the same mother twice 

284 if any(ancestor is el for el in alreadyChecked): 

285 continue 

286 alreadyChecked.append(ancestor) 

287 #Check if ancestor passes the group filter (if it has been covered/tested or not): 

288 if self.elementFilter(ancestor): 

289 continue 

290 #Subtract the weight of the ancestor and skip checking for all the older family tree of the ancestor 

291 #(avoid subtracting the same weight twice from mother and grandmother for instance) 

292 #(since the ancestorList is sorted by generation, the mother always 

293 #appears before the grandmother in the list) 

294 alreadyChecked += ancestor.getAncestors() 

295 if not hasattr(ancestor,'_totalXsec'): 

296 xsec = ancestor.weight.getXsecsFor(self.sqrts) 

297 ancestor._totalXsec = xsec[0].value.asNumber(fb) 

298 overlapXsec += ancestor._totalXsec 

299 

300 return missingX-overlapXsec

303 def getTotalXSec(self, sqrts=None): 

304 """ 

305 Calculate total missing topology cross section at sqrts. If no sqrts is given use self.sqrts 

306 :ivar sqrts: sqrts 

307 """ 

308 xsec = 0. 

309 if not sqrts: sqrts = self.sqrts 

310 for genEl in self.generalElements: 

311 xsec += genEl.missingX 

312 return xsec

314 def addToGeneralElements(self, el, missingX): 

315 """ 

316 Adds an element to the list of missing topologies = general elements. 

317 If the element contributes to a missing topology that is already in the list, add element and weight to topology. 

318 :parameter el: element to be added 

319 :parameter missingX: missing cross-section for the element (in fb) 

320 """ 

321 

322 newGenEl = GeneralElement(el,missingX,self.smFinalStates,self.bsmFinalStates) 

323 

324 index = index_bisect(self.generalElements,newGenEl) 

325 if index != len(self.generalElements) and self.generalElements[index] == newGenEl: 

326 self.generalElements[index]._contributingElements.append(el) 

327 self.generalElements[index].missingX += missingX 

328 else: 

329 self.generalElements.insert(index,newGenEl)

332class GeneralElement(object): 

333 """ 

334 This class represents a simplified (general) element which does 

335 only holds information about its even particles and decay type. 

336 The even particles are replaced/grouped by the particles defined in smFinalStates. 

337 """ 

338 

339 def __init__(self,el,missingX,smFinalStates,bsmFinalStates): 

340 

341 self.branches = [Branch() for _ in el.branches] 

342 self.missingX = missingX 

343 self.finalBSMstates = [] 

344 

345 for ib,branch in enumerate(el.branches): 

346 newParticles = [] 

347 for vertex in branch.evenParticles: 

348 newVertex = vertex[:] 

349 for ip,particle in enumerate(vertex): 

350 for particleList in smFinalStates: 

351 if particleList.contains(particle): 

352 newVertex[ip] = particleList 

353 newParticles.append(ParticleList(newVertex)) 

354 self.branches[ib].evenParticles = newParticles 

355 

356 finalBSM = branch.oddParticles[-1] 

357 for bsmFS in bsmFinalStates: 

358 if finalBSM == bsmFS: 

359 finalBSM = bsmFS 

360 break 

361 self.branches[ib].finalBSMstate = finalBSM 

362 self.branches[ib].setInfo() 

363 

364 self.sortBranches() 

365 self.finalBSMstates = [br.finalBSMstate for br in self.branches] 

366 self._contributingElements = [el] 

367 self._outputDescription = str(self)

369 def __cmp__(self,other): 

370 """ 

371 Compares the element with other for any branch ordering. 

372 The comparison is made based on branches. 

373 OBS: The elements and the branches must be sorted! 

374 :param other: element to be compared (GeneralElement object) 

375 :return: -1 if self < other, 0 if self == other, +1, if self > other. 

376 """ 

377 

378 if not isinstance(other,GeneralElement): 

379 return -1 

380 

381 #Compare branches: 

382 if self.branches != other.branches: 

383 comp = (self.branches > other.branches) 

384 if comp: 

385 return 1 

386 else: 

387 return -1 

388 #Compare decay type 

389 if self.finalBSMstates != other.finalBSMstates: 

390 comp = (self.finalBSMstates > other.finalBSMstates) 

391 if comp: 

392 return 1 

393 else: 

394 return -1 

395 

396 return 0

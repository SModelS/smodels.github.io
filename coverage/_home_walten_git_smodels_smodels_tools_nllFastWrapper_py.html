72 def _interpolateKfactors( self, kFacsVector, xval): 

73 """ 

74 Interpolate a list of k-factor values from 

75 kFacsVector = [[x0,[k1,k2,..]], [x1,[k1,k2,..],...]. 

76 FIXME what is xval? 

77 

78 :returns: list of interpolated k-factor values at x-value xval 

79 

80 """ 

81 import numpy 

82 kFacs = [] 

83 

84 xpts = [x[0] for x in kFacsVector] 

85 ypts = [x[1] for x in kFacsVector] 

86 coeffs = numpy.matrix.transpose(numpy.polyfit(xpts, ypts, len(xpts) - 1)) 

87 for ik in range(len(ypts[0])): 

88 kfac = 0. 

89 for ip, coeff in enumerate(coeffs[ik]): 

90 kfac += coeff * xval ** (len(xpts) - 1 - ip) 

91 if kfac <= 0.: 

92 kfac = 1. 

93 kFacs.append(kfac) 

94 

95 return kFacs

97 def _getKfactorsFrom( self, output ): 

98 """ 

99 Read NLLfast output and return the k-factors. 

100 

101 """ 

102 if not output: 

103 return False 

104 else: 

105 lines = output.split('\n') 

106 il = 0 

107 line = lines[il] 

108 process = False 

109 while not "K_NLO" in line and il < len(lines) - 2: 

110 if "process" in line: 

111 process = line[line.find("process:") + 8:].replace(" ", "") 

112 il += 1 

113 line = lines[il] 

114 if not process: 

115 return False 

116 # Line with header 

117 line = lines[il] 

118 # Count number of mass entries 

119 nmass = line.count('GeV') 

120 # Line with values 

121 line = lines[il + 2] 

122 data = [eval(x) for x in line.split()] 

123 if len(data) != nmass + 11: 

124 return False 

125 else: 

126 kFacs = data[9 + nmass:] 

127 

128 return kFacs

155 def _getProcessName(self, pIDs): 

156 """ 

157 Return the process name (in NLLfast notation) for the pair production of 

158 pIDs. 

159 

160 :returns: None, if the particle ID pair is not contained in NLLfast 

161 """ 

162 pid1, pid2 = sorted(pIDs) 

163 process = None 

164 

165 # Obtain the type of process: 

166 # - gluino-gluino production = gg 

167 # - squark-antisquark = sb 

168 # - squark-squark = ss 

169 # - squark-gluino = sg 

170 # - antistop-stop 

171 # - antisbottom-sbottom = st 

172 if pid1 in antisquarks and pid2 in squarks: 

173 process = 'sb' 

174 elif abs(pid1) in squarks and abs(pid2) in squarks: 

175 process = 'ss' 

176 elif pid1 == pid2 and pid1 in gluinos: 

177 process = 'gg' 

178 elif abs(pid1) in squarks and pid2 == 1000021 or \ 

179 abs(pid2) in squarks and pid1 == 1000021: 

180 process = 'sg' 

181 elif abs(pid1) == pid2 and pid2 in third: 

182 process = 'st' 

183 return process

185 def _getDecoupledKfactors( self, process, energy, pdf, mass ): 

186 """ 

187 Compute k-factors in the decoupled (squark or gluino) regime for the process. 

188 If a decoupled grid does not exist for the process, return None 

189 """ 

190 

191 if process != 'sb' and process != 'gg': return None 

192 elif process == 'sb': process_dcpl = 'sdcpl' 

193 elif process == 'gg': process_dcpl = 'gdcpl' 

194 nll_run = "./nllfast_" + energy + " %s %s %s" % \ 

195 (process_dcpl, pdf, mass) 

196 e = energy.replace ( "TeV", "" ).replace ( "*", "" ) 

197 # tool = toolBox.ToolBox().get ( "nllfast%d" % int ( e ) ) 

198 nll_output = self._run(nll_run ) 

199 if "K_NLO" in nll_output: 

200 return self._getKfactorsFrom(nll_output) 

201 else: return None

207 def getKfactorsFor( self, pIDs, slhafile, pdf='cteq' ): 

208 """ 

209 Read the NLLfast grid and returns a pair of k-factors (NLO and NLL) for  

210 the PIDs pair. 

211 

212 :returns: k-factors = None, if NLLfast does not contain the process; uses 

213 the slhafile to obtain the SUSY spectrum. 

214  

215 """ 

216 if not os.path.isfile(slhafile): 

217 logger.error("SLHA file %s not found", slhafile) 

218 return False 

219 

220 energy = str(int(self.sqrts)) + 'TeV' 

221 # Get process name (in NLLfast notation) 

222 process = self._getProcessName(pIDs) 

223 if not process: 

224 # Return k-factors = None, if NLLfast does not have the process 

225 return (None, None) 

226 

227 # Obtain relevant masses 

228 readfile = pyslha.readSLHAFile(slhafile) 

229 masses=readfile.blocks['MASS'] 

230 check_pids=squarks+gluinos+third 

231 for check in check_pids: 

232 if not check in masses.entries: 

233 logger.error ( "cannot compute k factor for pdgid %d: " \ 

234 " no particle mass given. will set mass to inf." % check ) 

235 masses.entries[check]=1.e10 

236 

237 gluinomass = abs(masses.entries[1000021]) 

238 squarkmass = sum([abs(masses.entries[pid]) 

239 for pid in squarks]) / 8. 

240 pid1, pid2 = sorted(pIDs) 

241 if pid1 in antisquarks and pid2 in squarks: 

242 squarkmass = (abs(masses.entries[abs(pid1)]) + 

243 abs(masses.entries[pid2])) / 2. 

244 elif pid1 in squarks and pid2 in squarks: 

245 squarkmass = (abs(masses.entries[pid1]) + abs(masses.entries[pid2])) / 2. 

246 elif abs(pid1) == pid2 and pid2 in third: 

247 squarkmass = abs(masses.entries[abs(pid1)]) 

248 

249 #if tool == None: 

250 # logger.warning("No NLLfast data for sqrts = " + str(sqrts)) 

251 # return (None, None) 

252 nllpath = self.installDirectory() 

253 # self.pathOfExecutable() 

254 self.checkInstallation() 

255 nll_output = self._compute ( energy, pIDs, pdf, squarkmass, gluinomass ) 

256 

257 # If run was successful, return k-factors: 

258 if "K_NLO" in nll_output: 

259 # NLLfast ran ok, try to get the k-factors 

260 kFacs = self._getKfactorsFrom(nll_output) 

261 if not kFacs or min(kFacs) <= 0.: 

262 logger.warning("Error obtaining k-factors") 

263 return (None, None) 

264 else: 

265 return kFacs 

266 # If run was not successful, check for decoupling error messages: 

267 elif not "too low/high" in nll_output.lower(): 

268 logger.warning("Error running NLLfast") 

269 return (None, None) 

270 

271 # Check for decoupling cases with a decoupling grid (only for sb and gg) 

272 doDecoupling = False 

273 if "too low/high gluino" in nll_output.lower(): 

274 if gluinomass > 500. and process == 'sb': 

275 doDecoupling = True 

276 dcpl_mass = gluinomass 

277 elif "too low/high squark" in nll_output.lower(): 

278 if squarkmass > 500. and process == 'gg': 

279 doDecoupling = True 

280 dcpl_mass = squarkmass 

281 

282 # If process do not have decoupled grids, return None: 

283 if not doDecoupling: 

284 if gluinomass < 99000. and squarkmass < 99000.: 

285 ## dont warn about obviously decoupled cases 

286 logger.warning("Masses of (q,g)=(%s,%s) out of NLLfast grid for %s, %s" % ( squarkmass, gluinomass, process, energy )) 

287 return (None, None) 

288 

289 # Obtain k-factors from the NLLfast decoupled grid 

290 kfacs = self._getDecoupledKfactors(process,energy,pdf,min(gluinomass,squarkmass)) 

291 # Decoupling limit is satisfied, do not interpolate 

292 if not kfacs: 

293 logger.warning("Error obtaining k-factors from the NLLfast decoupled grid for " + process) 

294 return (None, None) 

295 elif dcpl_mass/min(gluinomass,squarkmass) > 10.: 

296 return kfacs 

297 # Interpolate between the non-decoupled and decoupled grids 

298 else: 

299 kFacsVector = [[10.*min(gluinomass,squarkmass),kfacs]] #First point for interpolation (decoupled grid) 

300 kfacs = None 

301 while not kfacs and dcpl_mass > 500.: 

302 dcpl_mass -= 100. # Reduce decoupled mass, until NLLfast produces results 

303 if process == 'sb': nllinput = (process, pdf, squarkmass, dcpl_mass) 

304 else: nllinput = (process, pdf, dcpl_mass, gluinomass) 

305 nll_output = self._runForDecoupled ( energy, nllinput ) 

306 kfacs = self._getKfactorsFrom(nll_output) 

307 kFacsVector.append([dcpl_mass, kfacs]) #Second point for interpolation (non-decoupled grid) 

308 

309 if len(kFacsVector) < 2: 

310 logger.warning("Not enough points for interpolation in the decoupling " 

311 "limit") 

312 return (None, None) 

313 else: 

314 # Interpolate k-factors 

315 kFacs = self._interpolateKfactors(kFacsVector, 

316 max(squarkmass, gluinomass)) 

317 return kFacs

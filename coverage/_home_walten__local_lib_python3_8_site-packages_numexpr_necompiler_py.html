80class ASTNode(object): 

81 """Abstract Syntax Tree node. 

82 

83 Members: 

84 

85 astType -- type of node (op, constant, variable, raw, or alias) 

86 astKind -- the type of the result (bool, float, etc.) 

87 value -- value associated with this node. 

88 An opcode, numerical value, a variable name, etc. 

89 children -- the children below this node 

90 reg -- the register assigned to the result for this node. 

91 """ 

92 cmpnames = ['astType', 'astKind', 'value', 'children'] 

93 

94 def __init__(self, astType='generic', astKind='unknown', 

95 value=None, children=()): 

96 object.__init__(self) 

97 self.astType = astType 

98 self.astKind = astKind 

99 self.value = value 

100 self.children = tuple(children) 

101 self.reg = None 

102 

103 def __eq__(self, other): 

104 if self.astType == 'alias': 

105 self = self.value 

106 if other.astType == 'alias': 

107 other = other.value 

108 if not isinstance(other, ASTNode): 

109 return False 

110 for name in self.cmpnames: 

111 if getattr(self, name) != getattr(other, name): 

112 return False 

113 return True 

114 

115 def __lt__(self,other): 

116 # RAM: this is a fix for issue #88 whereby sorting on constants  

117 # that may be of astKind == 'complex' but type(self.value) == int or float 

118 # Here we let NumPy sort as it will cast data properly for comparison  

119 # when the Python built-ins will raise an error. 

120 if self.astType == 'constant': 

121 if self.astKind == other.astKind: 

122 return numpy.array(self.value) < numpy.array(other.value) 

123 return self.astKind < other.astKind 

124 else: 

125 raise TypeError( 'Sorting not implemented for astType: %s'%self.astType ) 

126 

127 def __hash__(self): 

128 if self.astType == 'alias': 

129 self = self.value 

130 return hash((self.astType, self.astKind, self.value, self.children)) 

131 

132 def __str__(self): 

133 return 'AST(%s, %s, %s, %s, %s)' % (self.astType, self.astKind, 

134 self.value, self.children, self.reg) 

135 

136 def __repr__(self): 

137 return '<AST object at %s>' % id(self) 

138 

139 def key(self): 

140 return (self.astType, self.astKind, self.value, self.children) 

141 

142 def typecode(self): 

143 return kind_to_typecode[self.astKind] 

144 

145 def postorderWalk(self): 

146 for c in self.children: 

147 for w in c.postorderWalk(): 

148 yield w 

149 yield self 

150 

151 def allOf(self, *astTypes): 

152 astTypes = set(astTypes) 

153 for w in self.postorderWalk(): 

154 if w.astType in astTypes: 

155 yield w

169def sigPerms(s): 

170 """Generate all possible signatures derived by upcasting the given 

171 signature. 

172 """ 

173 codes = 'bilfdc' 

174 if not s: 

175 yield '' 

176 elif s[0] in codes: 

177 start = codes.index(s[0]) 

178 for x in codes[start:]: 

179 for y in sigPerms(s[1:]): 

180 yield x + y 

181 elif s[0] == 's': # numbers shall not be cast to strings 

182 for y in sigPerms(s[1:]): 

183 yield 's' + y 

184 else: 

185 yield s

188def typeCompileAst(ast): 

189 """Assign appropiate types to each node in the AST. 

190 

191 Will convert opcodes and functions to appropiate upcast version, 

192 and add "cast" ops if needed. 

193 """ 

194 children = list(ast.children) 

195 if ast.astType == 'op': 

196 retsig = ast.typecode() 

197 basesig = ''.join(x.typecode() for x in list(ast.children)) 

198 # Find some operation that will work on an acceptable casting of args. 

199 for sig in sigPerms(basesig): 

200 value = (ast.value + '_' + retsig + sig).encode('ascii') 

201 if value in interpreter.opcodes: 

202 break 

203 else: 

204 for sig in sigPerms(basesig): 

205 funcname = (ast.value + '_' + retsig + sig).encode('ascii') 

206 if funcname in interpreter.funccodes: 

207 value = ('func_%sn' % (retsig + sig)).encode('ascii') 

208 children += [ASTNode('raw', 'none', 

209 interpreter.funccodes[funcname])] 

210 break 

211 else: 

212 raise NotImplementedError( 

213 "couldn't find matching opcode for '%s'" 

214 % (ast.value + '_' + retsig + basesig)) 

215 # First just cast constants, then cast variables if necessary: 

216 for i, (have, want) in enumerate(zip(basesig, sig)): 

217 if have != want: 

218 kind = typecode_to_kind[want] 

219 if children[i].astType == 'constant': 

220 children[i] = ASTNode('constant', kind, children[i].value) 

221 else: 

222 opname = "cast" 

223 children[i] = ASTNode('op', kind, opname, [children[i]]) 

224 else: 

225 value = ast.value 

226 children = ast.children 

227 return ASTNode(ast.astType, ast.astKind, value, 

228 [typeCompileAst(c) for c in children])

273def stringToExpression(s, types, context): 

274 """Given a string, convert it to a tree of ExpressionNode's. 

275 """ 

276 old_ctx = expressions._context.get_current_context() 

277 try: 

278 expressions._context.set_new_context(context) 

279 # first compile to a code object to determine the names 

280 if context.get('truediv', False): 

281 flags = __future__.division.compiler_flag 

282 else: 

283 flags = 0 

284 c = compile(s, '<expr>', 'eval', flags) 

285 # make VariableNode's for the names 

286 names = {} 

287 for name in c.co_names: 

288 if name == "None": 

289 names[name] = None 

290 elif name == "True": 

291 names[name] = True 

292 elif name == "False": 

293 names[name] = False 

294 else: 

295 t = types.get(name, default_type) 

296 names[name] = expressions.VariableNode(name, type_to_kind[t]) 

297 names.update(expressions.functions) 

298 # now build the expression 

299 ex = eval(c, names) 

300 if expressions.isConstant(ex): 

301 ex = expressions.ConstantNode(ex, expressions.getKind(ex)) 

302 elif not isinstance(ex, expressions.ExpressionNode): 

303 raise TypeError("unsupported expression type: %s" % type(ex)) 

304 finally: 

305 expressions._context.set_new_context(old_ctx) 

306 return ex

315 """Derive the input order of the variables in an expression. 

316 """ 

317 variables = {} 

318 for a in ast.allOf('variable'): 

319 variables[a.value] = a

321 

322 if input_order: 

323 if variable_names != set(input_order): 

324 raise ValueError( 

325 "input names (%s) don't match those found in expression (%s)"

328 ordered_names = input_order 

329 else:

331 ordered_names.sort()

387def collapseDuplicateSubtrees(ast): 

388 """Common subexpression elimination. 

389 """ 

390 seen = {} 

391 aliases = [] 

392 for a in ast.allOf('op'): 

393 if a in seen: 

394 target = seen[a] 

395 a.astType = 'alias' 

396 a.value = target 

397 a.children = () 

398 aliases.append(a) 

399 else: 

400 seen[a] = a 

401 # Set values and registers so optimizeTemporariesAllocation 

402 # doesn't get confused 

403 for a in aliases: 

404 while a.value.astType == 'alias': 

405 a.value = a.value.value 

406 return aliases

409def optimizeTemporariesAllocation(ast): 

410 """Attempt to minimize the number of temporaries needed, by 

411 reusing old ones. 

412 """ 

413 nodes = [n for n in ast.postorderWalk() if n.reg.temporary] 

414 users_of = dict((n.reg, set()) for n in nodes) 

415 

416 node_regs = dict((n, set(c.reg for c in n.children if c.reg.temporary)) 

417 for n in nodes) 

418 if nodes and nodes[-1] is not ast: 

419 nodes_to_check = nodes + [ast] 

420 else: 

421 nodes_to_check = nodes 

422 for n in nodes_to_check: 

423 for c in n.children: 

424 if c.reg.temporary: 

425 users_of[c.reg].add(n) 

426 

427 unused = dict([(tc, set()) for tc in scalar_constant_kinds]) 

428 for n in nodes: 

429 for c in n.children: 

430 reg = c.reg 

431 if reg.temporary: 

432 users = users_of[reg] 

433 users.discard(n) 

434 if not users: 

435 unused[reg.node.astKind].add(reg) 

436 if unused[n.astKind]: 

437 reg = unused[n.astKind].pop() 

438 users_of[reg] = users_of[n.reg] 

439 n.reg = reg

450def setRegisterNumbersForTemporaries(ast, start): 

451 """Assign register numbers for temporary registers, keeping track of 

452 aliases and handling immediate operands. 

453 """ 

454 seen = 0 

455 signature = '' 

456 aliases = [] 

457 for node in ast.postorderWalk(): 

458 if node.astType == 'alias': 

459 aliases.append(node) 

460 node = node.value 

461 if node.reg.immediate: 

462 node.reg.n = node.value 

463 continue 

464 reg = node.reg 

465 if reg.n is None: 

466 reg.n = start + seen 

467 seen += 1 

468 signature += reg.node.typecode() 

469 for node in aliases: 

470 node.reg = node.value.reg 

471 return start + seen, signature

474def convertASTtoThreeAddrForm(ast): 

475 """Convert an AST to a three address form. 

476 

477 Three address form is (op, reg1, reg2, reg3), where reg1 is the 

478 destination of the result of the instruction. 

479 

480 I suppose this should be called three register form, but three 

481 address form is found in compiler theory. 

482 """ 

483 return [(node.value, node.reg) + tuple([c.reg for c in node.children]) 

484 for node in ast.allOf('op')]

505 def quadrupleToString(opcode, store, a1=None, a2=None): 

506 cop = chr(interpreter.opcodes[opcode]).encode('ascii') 

507 cs = nToChr(store) 

508 ca1 = nToChr(a1) 

509 ca2 = nToChr(a2) 

510 return cop + cs + ca1 + ca2

536 d = kwargs.copy()

554def precompile(ex, signature=(), context={}): 

555 """Compile the expression to an intermediate form. 

556 """ 

557 types = dict(signature) 

558 input_order = [name for (name, type_) in signature] 

559 

560 if isinstance(ex, str): 

561 ex = stringToExpression(ex, types, context) 

562 

563 # the AST is like the expression, but the node objects don't have 

564 # any odd interpretations 

565 

566 ast = expressionToAST(ex) 

567 

568 if ex.astType != 'op': 

569 ast = ASTNode('op', value='copy', astKind=ex.astKind, children=(ast,)) 

570 

571 ast = typeCompileAst(ast) 

572 

573 aliases = collapseDuplicateSubtrees(ast) 

574 

575 assignLeafRegisters(ast.allOf('raw'), Immediate) 

576 assignLeafRegisters(ast.allOf('variable', 'constant'), Register) 

577 assignBranchRegisters(ast.allOf('op'), Register) 

578 

579 # assign registers for aliases 

580 for a in aliases: 

581 a.reg = a.value.reg 

582 

583 input_order = getInputOrder(ast, input_order) 

584 constants_order, constants = getConstants(ast) 

585 

586 if isReduction(ast): 

587 ast.reg.temporary = False 

588 

589 optimizeTemporariesAllocation(ast) 

590 

591 ast.reg.temporary = False 

592 r_output = 0 

593 ast.reg.n = 0 

594 

595 r_inputs = r_output + 1 

596 r_constants = setOrderedRegisterNumbers(input_order, r_inputs) 

597 r_temps = setOrderedRegisterNumbers(constants_order, r_constants) 

598 r_end, tempsig = setRegisterNumbersForTemporaries(ast, r_temps) 

599 

600 threeAddrProgram = convertASTtoThreeAddrForm(ast) 

601 input_names = tuple([a.value for a in input_order]) 

602 signature = ''.join(type_to_typecode[types.get(x, default_type)] 

603 for x in input_names) 

604 return threeAddrProgram, signature, tempsig, constants, input_names

607def NumExpr(ex, signature=(), **kwargs): 

608 """ 

609 Compile an expression built using E.<variable> variables to a function. 

610 

611 ex can also be specified as a string "2*a+3*b". 

612 

613 The order of the input variables and their types can be specified using the 

614 signature parameter, which is a list of (name, type) pairs. 

615 

616 Returns a `NumExpr` object containing the compiled function. 

617 """ 

618 # NumExpr can be called either directly by the end-user, in which case 

619 # kwargs need to be sanitized by getContext, or by evaluate, 

620 # in which case kwargs are in already sanitized. 

621 # In that case frame_depth is wrong (it should be 2) but it doesn't matter 

622 # since it will not be used (because truediv='auto' has already been 

623 # translated to either True or False). 

624 

625 context = getContext(kwargs, frame_depth=1) 

626 threeAddrProgram, inputsig, tempsig, constants, input_names = precompile(ex, signature, context) 

627 program = compileThreeAddrForm(threeAddrProgram) 

628 return interpreter.NumExpr(inputsig.encode('ascii'), 

629 tempsig.encode('ascii'), 

630 program, constants, input_names)

633def disassemble(nex): 

634 """ 

635 Given a NumExpr object, return a list which is the program disassembled. 

636 """ 

637 rev_opcodes = {} 

638 for op in interpreter.opcodes: 

639 rev_opcodes[interpreter.opcodes[op]] = op 

640 r_constants = 1 + len(nex.signature) 

641 r_temps = r_constants + len(nex.constants) 

642 

643 def getArg(pc, offset): 

644 if sys.version_info[0] < 3: 

645 arg = ord(nex.program[pc + offset]) 

646 op = rev_opcodes.get(ord(nex.program[pc])) 

647 else: 

648 arg = nex.program[pc + offset] 

649 op = rev_opcodes.get(nex.program[pc]) 

650 try: 

651 code = op.split(b'_')[1][offset - 1] 

652 except IndexError: 

653 return None 

654 if sys.version_info[0] > 2: 

655 # int.to_bytes is not available in Python < 3.2 

656 #code = code.to_bytes(1, sys.byteorder) 

657 code = bytes([code]) 

658 if arg == 255: 

659 return None 

660 if code != b'n': 

661 if arg == 0: 

662 return b'r0' 

663 elif arg < r_constants: 

664 return ('r%d[%s]' % (arg, nex.input_names[arg - 1])).encode('ascii') 

665 elif arg < r_temps: 

666 return ('c%d[%s]' % (arg, nex.constants[arg - r_constants])).encode('ascii') 

667 else: 

668 return ('t%d' % (arg,)).encode('ascii') 

669 else: 

670 return arg 

671 

672 source = [] 

673 for pc in range(0, len(nex.program), 4): 

674 if sys.version_info[0] < 3: 

675 op = rev_opcodes.get(ord(nex.program[pc])) 

676 else: 

677 op = rev_opcodes.get(nex.program[pc]) 

678 dest = getArg(pc, 1) 

679 arg1 = getArg(pc, 2) 

680 arg2 = getArg(pc, 3) 

681 source.append((op, dest, arg1, arg2)) 

682 return source

687 if kind == 'b': 

688 return bool 

689 if kind in 'iu': 

690 if a.dtype.itemsize > 4: 

691 return long_ # ``long`` is for integers of more than 32 bits 

692 if kind == 'u' and a.dtype.itemsize == 4: 

693 return long_ # use ``long`` here as an ``int`` is not enough 

694 return int_ 

695 if kind == 'f': 

696 if a.dtype.itemsize > 4: 

697 return double # ``double`` is for floats of more than 32 bits 

698 return float 

699 if kind == 'c': 

700 return complex 

701 if kind == 'S': 

702 return bytes 

703 raise ValueError("unknown type %s" % a.dtype.name)

706def getExprNames(text, context): 

707 ex = stringToExpression(text, {}, context) 

708 ast = expressionToAST(ex) 

709 input_order = getInputOrder(ast, None) 

710 #try to figure out if vml operations are used by expression 

711 if not use_vml: 

712 ex_uses_vml = False 

713 else: 

714 for node in ast.postorderWalk(): 

715 if node.astType == 'op' and node.value in vml_functions: 

716 ex_uses_vml = True 

717 break 

718 else: 

719 ex_uses_vml = False 

720 

721 return [a.value for a in input_order], ex_uses_vml

724def getArguments(names, local_dict=None, global_dict=None): 

725 """Get the arguments based on the names.""" 

726 call_frame = sys._getframe(2) 

727 

728 clear_local_dict = False 

729 if local_dict is None: 

730 local_dict = call_frame.f_locals 

731 clear_local_dict = True 

732 try: 

733 frame_globals = call_frame.f_globals 

734 if global_dict is None: 

735 global_dict = frame_globals 

736 

737 # If `call_frame` is the top frame of the interpreter we can't clear its  

738 # `local_dict`, because it is actually the `global_dict`. 

739 clear_local_dict = clear_local_dict and not frame_globals is local_dict 

740 

741 arguments = [] 

742 for name in names: 

743 try: 

744 a = local_dict[name] 

745 except KeyError: 

746 a = global_dict[name] 

747 arguments.append(numpy.asarray(a)) 

748 finally: 

749 # If we generated local_dict via an explicit reference to f_locals, 

750 # clear the dict to prevent creating extra ref counts in the caller's scope 

751 # See https://github.com/pydata/numexpr/issues/310 

752 if clear_local_dict: 

753 local_dict.clear() 

754 

755 return arguments

765def evaluate(ex, local_dict=None, global_dict=None, 

766 out=None, order='K', casting='safe', **kwargs): 

767 """Evaluate a simple array expression element-wise, using the new iterator. 

768 

769 ex is a string forming an expression, like "2*a+3*b". The values for "a" 

770 and "b" will by default be taken from the calling function's frame 

771 (through use of sys._getframe()). Alternatively, they can be specifed 

772 using the 'local_dict' or 'global_dict' arguments. 

773 

774 Parameters 

775 ---------- 

776 

777 local_dict : dictionary, optional 

778 A dictionary that replaces the local operands in current frame. 

779 

780 global_dict : dictionary, optional 

781 A dictionary that replaces the global operands in current frame. 

782 

783 out : NumPy array, optional 

784 An existing array where the outcome is going to be stored. Care is 

785 required so that this array has the same shape and type than the 

786 actual outcome of the computation. Useful for avoiding unnecessary 

787 new array allocations. 

788 

789 order : {'C', 'F', 'A', or 'K'}, optional 

790 Controls the iteration order for operands. 'C' means C order, 'F' 

791 means Fortran order, 'A' means 'F' order if all the arrays are 

792 Fortran contiguous, 'C' order otherwise, and 'K' means as close to 

793 the order the array elements appear in memory as possible. For 

794 efficient computations, typically 'K'eep order (the default) is 

795 desired. 

796 

797 casting : {'no', 'equiv', 'safe', 'same_kind', 'unsafe'}, optional 

798 Controls what kind of data casting may occur when making a copy or 

799 buffering. Setting this to 'unsafe' is not recommended, as it can 

800 adversely affect accumulations. 

801 

802 * 'no' means the data types should not be cast at all. 

803 * 'equiv' means only byte-order changes are allowed. 

804 * 'safe' means only casts which can preserve values are allowed. 

805 * 'same_kind' means only safe casts or casts within a kind, 

806 like float64 to float32, are allowed. 

807 * 'unsafe' means any data conversions may be done. 

808 """ 

809 global _numexpr_last 

810 if not isinstance(ex, str): 

811 raise ValueError("must specify expression as a string") 

812 # Get the names for this expression 

813 context = getContext(kwargs, frame_depth=1) 

814 expr_key = (ex, tuple(sorted(context.items()))) 

815 if expr_key not in _names_cache: 

816 _names_cache[expr_key] = getExprNames(ex, context) 

817 names, ex_uses_vml = _names_cache[expr_key] 

818 arguments = getArguments(names, local_dict, global_dict) 

819 

820 # Create a signature 

821 signature = [(name, getType(arg)) for (name, arg) in 

822 zip(names, arguments)] 

823 

824 # Look up numexpr if possible. 

825 numexpr_key = expr_key + (tuple(signature),) 

826 try: 

827 compiled_ex = _numexpr_cache[numexpr_key] 

828 except KeyError: 

829 compiled_ex = _numexpr_cache[numexpr_key] = NumExpr(ex, signature, **context) 

830 kwargs = {'out': out, 'order': order, 'casting': casting, 

831 'ex_uses_vml': ex_uses_vml} 

832 _numexpr_last = dict(ex=compiled_ex, argnames=names, kwargs=kwargs) 

833 with evaluate_lock: 

834 return compiled_ex(*arguments, **kwargs)

837def re_evaluate(local_dict=None): 

838 """Re-evaluate the previous executed array expression without any check. 

839 

840 This is meant for accelerating loops that are re-evaluating the same 

841 expression repeatedly without changing anything else than the operands. 

842 If unsure, use evaluate() which is safer. 

843 

844 Parameters 

845 ---------- 

846 

847 local_dict : dictionary, optional 

848 A dictionary that replaces the local operands in current frame. 

849 

850 """ 

851 try: 

852 compiled_ex = _numexpr_last['ex'] 

853 except KeyError: 

854 raise RuntimeError("not a previous evaluate() execution found") 

855 argnames = _numexpr_last['argnames'] 

856 args = getArguments(argnames, local_dict) 

857 kwargs = _numexpr_last['kwargs'] 

858 with evaluate_lock: 

859 return compiled_ex(*args, **kwargs)

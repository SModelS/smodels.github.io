31def get_home_dir(): 

32 """Get the real path of the home directory""" 

33 homedir = os.path.expanduser('~') 

34 # Next line will make things work even when /home/ is a symlink to 

35 # /usr/home as it is on FreeBSD, for example 

36 homedir = os.path.realpath(homedir) 

37 homedir = py3compat.str_to_unicode(homedir, encoding=sys.getfilesystemencoding()) 

38 return homedir

41def _mkdtemp_once(name): 

42 """Make or reuse a temporary directory. 

43 

44 If this is called with the same name in the same process, it will return 

45 the same directory. 

46 """ 

47 try: 

48 return _dtemps[name] 

49 except KeyError: 

50 d = _dtemps[name] = tempfile.mkdtemp(prefix=name + '-') 

51 return d

53def jupyter_config_dir(): 

54 """Get the Jupyter config directory for this platform and user. 

55 

56 Returns JUPYTER_CONFIG_DIR if defined, else ~/.jupyter 

57 """ 

58 

59 env = os.environ 

60 home_dir = get_home_dir() 

61 

62 if env.get('JUPYTER_NO_CONFIG'): 

63 return _mkdtemp_once('jupyter-clean-cfg') 

64 

65 if env.get('JUPYTER_CONFIG_DIR'): 

66 return env['JUPYTER_CONFIG_DIR'] 

67 

68 return pjoin(home_dir, '.jupyter')

71def jupyter_data_dir(): 

72 """Get the config directory for Jupyter data files. 

73 

74 These are non-transient, non-configuration files. 

75 

76 Returns JUPYTER_DATA_DIR if defined, else a platform-appropriate path. 

77 """ 

78 env = os.environ 

79 

80 if env.get('JUPYTER_DATA_DIR'): 

81 return env['JUPYTER_DATA_DIR'] 

82 

83 home = get_home_dir() 

84 

85 if sys.platform == 'darwin': 

86 return os.path.join(home, 'Library', 'Jupyter') 

87 elif os.name == 'nt': 

88 appdata = os.environ.get('APPDATA', None) 

89 if appdata: 

90 return pjoin(appdata, 'jupyter') 

91 else: 

92 return pjoin(jupyter_config_dir(), 'data') 

93 else: 

94 # Linux, non-OS X Unix, AIX, etc. 

95 xdg = env.get("XDG_DATA_HOME", None) 

96 if not xdg: 

97 xdg = pjoin(home, '.local', 'share') 

98 return pjoin(xdg, 'jupyter')

132def jupyter_path(*subdirs): 

133 """Return a list of directories to search for data files 

134 

135 JUPYTER_PATH environment variable has highest priority. 

136 

137 If ``*subdirs`` are given, that subdirectory will be added to each element. 

138 

139 Examples: 

140 

141 >>> jupyter_path() 

142 ['~/.local/jupyter', '/usr/local/share/jupyter'] 

143 >>> jupyter_path('kernels') 

144 ['~/.local/jupyter/kernels', '/usr/local/share/jupyter/kernels'] 

145 """ 

146 

147 paths = [] 

148 # highest priority is env 

149 if os.environ.get('JUPYTER_PATH'): 

150 paths.extend( 

151 p.rstrip(os.sep) 

152 for p in os.environ['JUPYTER_PATH'].split(os.pathsep) 

153 ) 

154 # then user dir 

155 paths.append(jupyter_data_dir()) 

156 # then sys.prefix 

157 for p in ENV_JUPYTER_PATH: 

158 if p not in SYSTEM_JUPYTER_PATH: 

159 paths.append(p) 

160 # finally, system 

161 paths.extend(SYSTEM_JUPYTER_PATH) 

162 

163 # add subdir, if requested 

164 if subdirs: 

165 paths = [ pjoin(p, *subdirs) for p in paths ] 

166 return paths 

167 

168 

169if os.name == 'nt': 

170 programdata = os.environ.get('PROGRAMDATA', None) 

171 if programdata: 

172 SYSTEM_CONFIG_PATH = [os.path.join(programdata, 'jupyter')] 

173 else: # PROGRAMDATA is not defined by default on XP. 

174 SYSTEM_CONFIG_PATH = [] 

175else: 

176 SYSTEM_CONFIG_PATH = [ 

177 "/usr/local/etc/jupyter", 

178 "/etc/jupyter", 

179 ] 

180 

181ENV_CONFIG_PATH = [os.path.join('/etc', 'jupyter')]

216def is_file_hidden_win(abs_path, stat_res=None): 

217 """Is a file hidden? 

218 

219 This only checks the file itself; it should be called in combination with 

220 checking the directory containing the file. 

221 

222 Use is_hidden() instead to check the file and its parent directories. 

223 

224 Parameters 

225 ---------- 

226 abs_path : unicode 

227 The absolute path to check. 

228 stat_res : os.stat_result, optional 

229 Ignored on Windows, exists for compatibility with POSIX version of the 

230 function. 

231 """ 

232 if os.path.basename(abs_path).startswith('.'): 

233 return True 

234 

235 win32_FILE_ATTRIBUTE_HIDDEN = 0x02 

236 import ctypes 

237 try: 

238 attrs = ctypes.windll.kernel32.GetFileAttributesW( 

239 py3compat.cast_unicode(abs_path) 

240 ) 

241 except AttributeError: 

242 pass 

243 else: 

244 if attrs > 0 and attrs & win32_FILE_ATTRIBUTE_HIDDEN: 

245 return True 

246 

247 return False

250def is_file_hidden_posix(abs_path, stat_res=None): 

251 """Is a file hidden? 

252 

253 This only checks the file itself; it should be called in combination with 

254 checking the directory containing the file. 

255 

256 Use is_hidden() instead to check the file and its parent directories. 

257 

258 Parameters 

259 ---------- 

260 abs_path : unicode 

261 The absolute path to check. 

262 stat_res : os.stat_result, optional 

263 The result of calling stat() on abs_path. If not passed, this function 

264 will call stat() internally. 

265 """ 

266 if os.path.basename(abs_path).startswith('.'): 

267 return True 

268 

269 if stat_res is None or stat.S_ISLNK(stat_res.st_mode): 

270 try: 

271 stat_res = os.stat(abs_path) 

272 except OSError as e: 

273 if e.errno == errno.ENOENT: 

274 return False 

275 raise 

276 

277 # check that dirs can be listed 

278 if stat.S_ISDIR(stat_res.st_mode): 

279 # use x-access, not actual listing, in case of slow/large listings 

280 if not os.access(abs_path, os.X_OK | os.R_OK): 

281 return True 

282 

283 # check UF_HIDDEN 

284 if getattr(stat_res, 'st_flags', 0) & UF_HIDDEN: 

285 return True 

286 

287 return False 

288 

289 

290if sys.platform == 'win32': 

291 is_file_hidden = is_file_hidden_win 

292else: 

293 is_file_hidden = is_file_hidden_posix

296def is_hidden(abs_path, abs_root=''): 

297 """Is a file hidden or contained in a hidden directory? 

298 

299 This will start with the rightmost path element and work backwards to the 

300 given root to see if a path is hidden or in a hidden directory. Hidden is 

301 determined by either name starting with '.' or the UF_HIDDEN flag as 

302 reported by stat. 

303 

304 If abs_path is the same directory as abs_root, it will be visible even if 

305 that is a hidden folder. This only checks the visibility of files 

306 and directories *within* abs_root. 

307 

308 Parameters 

309 ---------- 

310 abs_path : unicode 

311 The absolute path to check for hidden directories. 

312 abs_root : unicode 

313 The absolute path of the root directory in which hidden directories 

314 should be checked for. 

315 """ 

316 if os.path.normpath(abs_path) == os.path.normpath(abs_root): 

317 return False 

318 

319 if is_file_hidden(abs_path): 

320 return True 

321 

322 if not abs_root: 

323 abs_root = abs_path.split(os.sep, 1)[0] + os.sep 

324 inside_root = abs_path[len(abs_root):] 

325 if any(part.startswith('.') for part in inside_root.split(os.sep)): 

326 return True 

327 

328 # check UF_HIDDEN on any location up to root. 

329 # is_file_hidden() already checked the file, so start from its parent dir 

330 path = os.path.dirname(abs_path) 

331 while path and path.startswith(abs_root) and path != abs_root: 

332 if not exists(path): 

333 path = os.path.dirname(path) 

334 continue 

335 try: 

336 # may fail on Windows junctions 

337 st = os.lstat(path) 

338 except OSError: 

339 return True 

340 if getattr(st, 'st_flags', 0) & UF_HIDDEN: 

341 return True 

342 path = os.path.dirname(path) 

343 

344 return False

347def win32_restrict_file_to_user(fname): 

348 """Secure a windows file to read-only access for the user. 

349 Follows guidance from win32 library creator: 

350 http://timgolden.me.uk/python/win32_how_do_i/add-security-to-a-file.html 

351 

352 This method should be executed against an already generated file which 

353 has no secrets written to it yet. 

354 

355 Parameters 

356 ---------- 

357 

358 fname : unicode 

359 The path to the file to secure 

360 """ 

361 import win32api 

362 import win32security 

363 import ntsecuritycon as con 

364 

365 # everyone, _domain, _type = win32security.LookupAccountName("", "Everyone") 

366 admins = win32security.CreateWellKnownSid(win32security.WinBuiltinAdministratorsSid) 

367 user, _domain, _type = win32security.LookupAccountName("", win32api.GetUserNameEx(win32api.NameSamCompatible)) 

368 

369 sd = win32security.GetFileSecurity(fname, win32security.DACL_SECURITY_INFORMATION) 

370 

371 dacl = win32security.ACL() 

372 # dacl.AddAccessAllowedAce(win32security.ACL_REVISION, con.FILE_ALL_ACCESS, everyone) 

373 dacl.AddAccessAllowedAce(win32security.ACL_REVISION, con.FILE_GENERIC_READ | con.FILE_GENERIC_WRITE, user) 

374 dacl.AddAccessAllowedAce(win32security.ACL_REVISION, con.FILE_ALL_ACCESS, admins) 

375 

376 sd.SetSecurityDescriptorDacl(1, dacl, 0) 

377 win32security.SetFileSecurity(fname, win32security.DACL_SECURITY_INFORMATION, sd)

380def get_file_mode(fname): 

381 """Retrieves the file mode corresponding to fname in a filesystem-tolerant manner. 

382 

383 Parameters 

384 ---------- 

385 

386 fname : unicode 

387 The path to the file to get mode from 

388 

389 """ 

390 # Some filesystems (e.g., CIFS) auto-enable the execute bit on files. As a result, we 

391 # should tolerate the execute bit on the file's owner when validating permissions - thus 

392 # the missing least significant bit on the third octal digit. In addition, we also tolerate 

393 # the sticky bit being set, so the lsb from the fourth octal digit is also removed. 

394 return stat.S_IMODE(os.stat(fname).st_mode) & 0o6677 # Use 4 octal digits since S_IMODE does the same

401def secure_write(fname, binary=False): 

402 """Opens a file in the most restricted pattern available for 

403 writing content. This limits the file mode to `0o0600` and yields 

404 the resulting opened filed handle. 

405 

406 Parameters 

407 ---------- 

408 

409 fname : unicode 

410 The path to the file to write 

411 

412 binary: boolean 

413 Indicates that the file is binary 

414 """ 

415 mode = 'wb' if binary else 'w' 

416 open_flag = os.O_CREAT | os.O_WRONLY | os.O_TRUNC 

417 try: 

418 os.remove(fname) 

419 except (IOError, OSError): 

420 # Skip any issues with the file not existing 

421 pass 

422 

423 if os.name == 'nt': 

424 if allow_insecure_writes: 

425 # Mounted file systems can have a number of failure modes inside this block. 

426 # For windows machines in insecure mode we simply skip this to avoid failures :/ 

427 issue_insecure_write_warning() 

428 else: 

429 # Python on windows does not respect the group and public bits for chmod, so we need 

430 # to take additional steps to secure the contents. 

431 # Touch file pre-emptively to avoid editing permissions in open files in Windows 

432 fd = os.open(fname, open_flag, 0o0600) 

433 os.close(fd) 

434 open_flag = os.O_WRONLY | os.O_TRUNC 

435 win32_restrict_file_to_user(fname) 

436 

437 with os.fdopen(os.open(fname, open_flag, 0o0600), mode) as f: 

438 if os.name != 'nt': 

439 # Enforce that the file got the requested permissions before writing 

440 file_mode = get_file_mode(fname) 

441 if 0o0600 != file_mode: 

442 if allow_insecure_writes: 

443 issue_insecure_write_warning() 

444 else: 

445 raise RuntimeError("Permissions assignment failed for secure file: '{file}'." 

446 " Got '{permissions}' instead of '0o0600'." 

447 .format(file=fname, permissions=oct(file_mode))) 

448 yield f

1"""Bridges between the `asyncio` module and Tornado IOLoop. 

2 

3.. versionadded:: 3.2 

4 

5This module integrates Tornado with the ``asyncio`` module introduced 

6in Python 3.4. This makes it possible to combine the two libraries on 

7the same event loop. 

8 

9.. deprecated:: 5.0 

10 

11 While the code in this module is still used, it is now enabled 

12 automatically when `asyncio` is available, so applications should 

13 no longer need to refer to this module directly. 

14 

15.. note:: 

16 

17 Tornado requires the `~asyncio.AbstractEventLoop.add_reader` family of 

18 methods, so it is not compatible with the `~asyncio.ProactorEventLoop` on 

19 Windows. Use the `~asyncio.SelectorEventLoop` instead. 

20"""

32class BaseAsyncIOLoop(IOLoop): 

33 def initialize(self, asyncio_loop, **kwargs): 

34 self.asyncio_loop = asyncio_loop 

35 # Maps fd to (fileobj, handler function) pair (as in IOLoop.add_handler) 

36 self.handlers = {} 

37 # Set of fds listening for reads/writes 

38 self.readers = set() 

39 self.writers = set() 

40 self.closing = False 

41 # If an asyncio loop was closed through an asyncio interface 

42 # instead of IOLoop.close(), we'd never hear about it and may 

43 # have left a dangling reference in our map. In case an 

44 # application (or, more likely, a test suite) creates and 

45 # destroys a lot of event loops in this way, check here to 

46 # ensure that we don't have a lot of dead loops building up in 

47 # the map. 

48 # 

49 # TODO(bdarnell): consider making self.asyncio_loop a weakref 

50 # for AsyncIOMainLoop and make _ioloop_for_asyncio a 

51 # WeakKeyDictionary. 

52 for loop in list(IOLoop._ioloop_for_asyncio): 

53 if loop.is_closed(): 

54 del IOLoop._ioloop_for_asyncio[loop] 

55 IOLoop._ioloop_for_asyncio[asyncio_loop] = self 

56 super(BaseAsyncIOLoop, self).initialize(**kwargs) 

57 

58 def close(self, all_fds=False): 

59 self.closing = True 

60 for fd in list(self.handlers): 

61 fileobj, handler_func = self.handlers[fd] 

62 self.remove_handler(fd) 

63 if all_fds: 

64 self.close_fd(fileobj) 

65 # Remove the mapping before closing the asyncio loop. If this 

66 # happened in the other order, we could race against another 

67 # initialize() call which would see the closed asyncio loop, 

68 # assume it was closed from the asyncio side, and do this 

69 # cleanup for us, leading to a KeyError. 

70 del IOLoop._ioloop_for_asyncio[self.asyncio_loop] 

71 self.asyncio_loop.close() 

72 

73 def add_handler(self, fd, handler, events): 

74 fd, fileobj = self.split_fd(fd) 

75 if fd in self.handlers: 

76 raise ValueError("fd %s added twice" % fd) 

77 self.handlers[fd] = (fileobj, stack_context.wrap(handler)) 

78 if events & IOLoop.READ: 

79 self.asyncio_loop.add_reader( 

80 fd, self._handle_events, fd, IOLoop.READ) 

81 self.readers.add(fd) 

82 if events & IOLoop.WRITE: 

83 self.asyncio_loop.add_writer( 

84 fd, self._handle_events, fd, IOLoop.WRITE) 

85 self.writers.add(fd) 

86 

87 def update_handler(self, fd, events): 

88 fd, fileobj = self.split_fd(fd) 

89 if events & IOLoop.READ: 

90 if fd not in self.readers: 

91 self.asyncio_loop.add_reader( 

92 fd, self._handle_events, fd, IOLoop.READ) 

93 self.readers.add(fd) 

94 else: 

95 if fd in self.readers: 

96 self.asyncio_loop.remove_reader(fd) 

97 self.readers.remove(fd) 

98 if events & IOLoop.WRITE: 

99 if fd not in self.writers: 

100 self.asyncio_loop.add_writer( 

101 fd, self._handle_events, fd, IOLoop.WRITE) 

102 self.writers.add(fd) 

103 else: 

104 if fd in self.writers: 

105 self.asyncio_loop.remove_writer(fd) 

106 self.writers.remove(fd) 

107 

108 def remove_handler(self, fd): 

109 fd, fileobj = self.split_fd(fd) 

110 if fd not in self.handlers: 

111 return 

112 if fd in self.readers: 

113 self.asyncio_loop.remove_reader(fd) 

114 self.readers.remove(fd) 

115 if fd in self.writers: 

116 self.asyncio_loop.remove_writer(fd) 

117 self.writers.remove(fd) 

118 del self.handlers[fd] 

119 

120 def _handle_events(self, fd, events): 

121 fileobj, handler_func = self.handlers[fd] 

122 handler_func(fileobj, events) 

123 

124 def start(self): 

125 try: 

126 old_loop = asyncio.get_event_loop() 

127 except (RuntimeError, AssertionError): 

128 old_loop = None 

129 try: 

130 self._setup_logging() 

131 asyncio.set_event_loop(self.asyncio_loop) 

132 self.asyncio_loop.run_forever() 

133 finally: 

134 asyncio.set_event_loop(old_loop) 

135 

136 def stop(self): 

137 self.asyncio_loop.stop() 

138 

139 def call_at(self, when, callback, *args, **kwargs): 

140 # asyncio.call_at supports *args but not **kwargs, so bind them here. 

141 # We do not synchronize self.time and asyncio_loop.time, so 

142 # convert from absolute to relative. 

143 return self.asyncio_loop.call_later( 

144 max(0, when - self.time()), self._run_callback, 

145 functools.partial(stack_context.wrap(callback), *args, **kwargs)) 

146 

147 def remove_timeout(self, timeout): 

148 timeout.cancel() 

149 

150 def add_callback(self, callback, *args, **kwargs): 

151 try: 

152 self.asyncio_loop.call_soon_threadsafe( 

153 self._run_callback, 

154 functools.partial(stack_context.wrap(callback), *args, **kwargs)) 

155 except RuntimeError: 

156 # "Event loop is closed". Swallow the exception for 

157 # consistency with PollIOLoop (and logical consistency 

158 # with the fact that we can't guarantee that an 

159 # add_callback that completes without error will 

160 # eventually execute). 

161 pass 

162 

163 add_callback_from_signal = add_callback

195class AsyncIOLoop(BaseAsyncIOLoop): 

196 """``AsyncIOLoop`` is an `.IOLoop` that runs on an ``asyncio`` event loop. 

197 This class follows the usual Tornado semantics for creating new 

198 ``IOLoops``; these loops are not necessarily related to the 

199 ``asyncio`` default event loop. 

200 

201 Each ``AsyncIOLoop`` creates a new ``asyncio.EventLoop``; this object 

202 can be accessed with the ``asyncio_loop`` attribute. 

203 

204 .. versionchanged:: 5.0 

205 

206 When an ``AsyncIOLoop`` becomes the current `.IOLoop`, it also sets 

207 the current `asyncio` event loop. 

208 

209 .. deprecated:: 5.0 

210 

211 Now used automatically when appropriate; it is no longer necessary 

212 to refer to this class directly. 

213 """ 

214 def initialize(self, **kwargs): 

215 self.is_current = False 

216 loop = asyncio.new_event_loop() 

217 try: 

218 super(AsyncIOLoop, self).initialize(loop, **kwargs) 

219 except Exception: 

220 # If initialize() does not succeed (taking ownership of the loop), 

221 # we have to close it. 

222 loop.close() 

223 raise 

224 

225 def close(self, all_fds=False): 

226 if self.is_current: 

227 self.clear_current() 

228 super(AsyncIOLoop, self).close(all_fds=all_fds)

230 def make_current(self): 

231 if not self.is_current: 

232 try: 

233 self.old_asyncio = asyncio.get_event_loop() 

234 except (RuntimeError, AssertionError):

240 if self.is_current:

242 self.is_current = False

273class AnyThreadEventLoopPolicy(asyncio.DefaultEventLoopPolicy): 

274 """Event loop policy that allows loop creation on any thread. 

275 

276 The default `asyncio` event loop policy only automatically creates 

277 event loops in the main threads. Other threads must create event 

278 loops explicitly or `asyncio.get_event_loop` (and therefore 

279 `.IOLoop.current`) will fail. Installing this policy allows event 

280 loops to be created automatically on any thread, matching the 

281 behavior of Tornado versions prior to 5.0 (or 5.0 on Python 2). 

282 

283 Usage:: 

284 

285 asyncio.set_event_loop_policy(AnyThreadEventLoopPolicy()) 

286 

287 .. versionadded:: 5.0 

288 

289 """ 

290 def get_event_loop(self): 

291 try: 

292 return super().get_event_loop() 

293 except (RuntimeError, AssertionError): 

294 # This was an AssertionError in python 3.4.2 (which ships with debian jessie) 

295 # and changed to a RuntimeError in 3.4.3. 

296 # "There is no current event loop in thread %r" 

297 loop = self.new_event_loop() 

298 self.set_event_loop(loop) 

299 return loop

75 def addHigherOrders ( self, sqrts, slhafile ): 

76 """ add higher order xsecs """ 

77 xsecs = copy.deepcopy ( self.loXsecs ) 

78 wlabel = str(int(sqrts / TeV)) + ' TeV' 

79 if self.maxOrder == LO: 

80 wlabel += ' (LO)' 

81 elif self.maxOrder == NLO: 

82 wlabel += ' (NLO)' 

83 elif self.maxOrder == NLL: 

84 wlabel += ' (NLO+NLL)' 

85 for ixsec, xsec in enumerate(xsecs): 

86 xsecs[ixsec].info.label = wlabel 

87 xsecs[ixsec].info.order = self.maxOrder 

88 

89 if self.maxOrder > 0: 

90 pIDs = self.loXsecs.getPIDpairs() 

91 nllfast = toolBox.ToolBox().get("nllfast%d" % sqrts.asNumber(TeV) ) 

92 for pID in pIDs: 

93 k = 0. 

94 kNLO, kNLL = nllfast.getKfactorsFor(pID, slhafile) 

95 if self.maxOrder == NLO and kNLO: 

96 k = kNLO 

97 elif self.maxOrder == NLL and kNLL and kNLO: 

98 k = kNLO * kNLL 

99 elif self.maxOrder > 2 and kNLL and kNLO: 

100 logger.warning("Unkown xsec order, using NLL+NLO k-factor, " 

101 "if available") 

102 k = kNLO * kNLL 

103 k = float(k) 

104 for i, xsec in enumerate(xsecs): 

105 if set(xsec.pid) == set(pID): 

106 # Apply k-factor 

107 xsecs[i] = xsec * k 

108 

109 # Remove zero cross sections 

110 while len(xsecs) > 0 and xsecs.getMinXsec() == 0. * pb: 

111 for xsec in xsecs: 

112 if xsec.value == 0. * pb: 

113 xsecs.delete(xsec) 

114 break 

115 if self.maxOrder > 0 and len(xsecs) == 0: 

116 self.countNoNLOXSecs += 1 

117 if self.countNoNLOXSecs < 3: 

118 logger.warning("No NLO or NLL cross sections available.") 

119 if self.countNoNLOXSecs == 3: 

120 logger.warning("No NLO or NLL cross sections available (will quench such warnings in future).") 

121 

122 #for i in xsecs: 

123 # logger.error ( "xsec=%s (%s)" % (i,type(i)) ) 

124 return xsecs

194 def compute ( self, sqrts, slhafile, lhefile=None, unlink=True, loFromSlha=None, 

195 pythiacard=None, ssmultipliers=None ): 

196 """ 

197 Run pythia and compute SUSY cross sections for the input SLHA file. 

198 

199 :param sqrts: sqrt{s} to run Pythia, given as a unum (e.g. 7.*TeV) 

200 :param slhafile: SLHA file 

201 :param lhefile: LHE file. If None, do not write pythia output to file. If 

202 file does not exist, write pythia output to this file name. If 

203 file exists, read LO xsecs from this file (does not run pythia). 

204 :param unlink: Clean up temp directory after running pythia 

205 :param loFromSlha: If True, uses the LO xsecs from the SLHA file to 

206 compute the higher order xsecs 

207 :param pythiaCard: Optional path to pythia.card. If None, uses 

208 smodels/etc/pythia.card 

209 :param ssmultipliers: optionally supply signal strengh multipliers, 

210 given as dictionary of the tuple of the mothers' pids as keys and 

211 multipliers as values, e.g { (1000001,1000021):1.1 }. 

212 :returns: XSectionList object 

213 

214 """ 

215 sqrts = self._checkSqrts( sqrts ) 

216 self._checkSLHA ( slhafile ) 

217 

218 if lhefile: 

219 if os.path.isfile(lhefile): 

220 logger.warning("Using LO cross sections from " + lhefile) 

221 logger.error ( "Cross section retrieval from lhefile currently not implemented" ) 

222 sys.exit() 

223 else: 

224 logger.info("Writing pythia LHE output to " + lhefile) 

225 if loFromSlha: 

226 logger.info("Using LO cross sections from " + slhafile) 

227 xsecsInfile = crossSection.getXsecFromSLHAFile(slhafile) 

228 loXsecs = crossSection.XSectionList() 

229 for xsec in xsecsInfile: 

230 if xsec.info.order == 0 and xsec.info.sqrts == sqrts: 

231 loXsecs.add(xsec) 

232 if ssmultipliers != None: 

233 logger.warning ( "supplied signal strength multipliers, but LO xsecs are taken from file. Will not apply them" ) 

234 else: 

235 logger.info("get LO cross sections from pythia%d" % self.pythiaVersion ) 

236 tool = toolBox.ToolBox().get("pythia%d" % self.pythiaVersion ) 

237 tool.nevents = self.nevents 

238 tool.sqrts = sqrts / TeV 

239 tool.pythiacard = pythiacard 

240 loXsecs = tool.run( slhafile, lhefile, unlink=unlink ) 

241 if ssmultipliers != None: 

242 loXsecs = self.applyMultipliers ( loXsecs, ssmultipliers ) 

243 self.loXsecs = loXsecs 

244 self.loXsecs.sort() 

245 self.xsecs = self.addHigherOrders ( sqrts, slhafile ) 

246 self.xsecs.sort() 

247 #for xsec in self.loXsecs: 

248 # logger.debug ( "now writing out xsecs: %s" % xsec.value ) 

249 logger.debug ( "how many NLL xsecs? %d" % len(self.xsecs) ) 

250 return self.xsecs

253 lOfromSLHA, tofile, pythiacard = None, 

254 ssmultipliers = None, comment = None ): 

255 """ 

256 compute the cross sections for one file. 

257 :param sqrtses: list of sqrt{s} tu run pythia, as a unum (e.g. [7*TeV]) 

258 :param inputFile: input SLHA file to compute xsecs for 

259 :param unlink: if False, keep temporary files 

260 :param lofromSLHA: try to obtain LO xsecs from SLHA file itself 

261 :param tofile: False, True, "all": write results to file, if "all" also write  

262 lower xsecs to file. 

263 :param pythiacard: optionally supply your own runcard 

264 :param ssmultipliers: optionally supply signal strengh multipliers, 

265 given as dictionary of the tuple of the mothers' pids as keys and 

266 multipliers as values, e.g { (1000001,1000021):1.1 }. 

267 :param comment: an optional comment that gets added to the slha file. 

268 :returns: number of xsections that have been computed 

269 """ 

270 nXSecs = 0 ## count the xsecs we are adding 

271 if tofile: 

272 logger.info("Computing SLHA cross section from %s, adding to " 

273 "SLHA file." % inputFile ) 

274 complain = True ## dont complain about already existing xsecs, 

275 # if we were the ones writing them 

276 for s in sqrtses: 

277 ss = s*TeV 

278 self.compute( ss, inputFile, unlink= unlink, loFromSlha= lOfromSLHA, 

279 pythiacard=pythiacard, ssmultipliers = ssmultipliers ) 

280 if tofile == "all": 

281 xcomment = str(self.nevents)+" evts, pythia%d [pb]"%\ 

282 self.pythiaVersion 

283 nXSecs += self.addXSecToFile(self.loXsecs, inputFile, xcomment, complain ) 

284 complain = False 

285 xcomment = str(self.nevents)+" events, [pb], pythia%d for LO"%\ 

286 self.pythiaVersion 

287 if tofile != False: 

288 nXSecs += self.addXSecToFile( self.xsecs, inputFile, xcomment, complain) 

289 complain = False 

290 if nXSecs > 0: ## only add if we actually added xsecs 

291 self.addMultipliersToFile ( ssmultipliers, inputFile ) 

292 self.addCommentToFile ( comment, inputFile ) 

293 else: 

294 logger.info("Computing SLHA cross section from %s." % inputFile ) 

295 print() 

296 print( " Cross sections:" ) 

297 print( "=======================" ) 

298 for s in sqrtses: 

299 ss = s*TeV 

300 self.compute( ss, inputFile, unlink=unlink, loFromSlha=lOfromSLHA, 

301 ssmultipliers = ssmultipliers ) 

302 for xsec in self.xsecs: 

303 nXSecs += 1 

304 print( "%s %20s: %.3e pb" % \ 

305 ( xsec.info.label,xsec.pid,xsec.value/pb ) ) 

306 print() 

307 return nXSecs

364 def addXSecToFile( self, xsecs, slhafile, comment=None, complain=True): 

365 """ 

366 Write cross sections to an SLHA file. 

367 

368 :param xsecs: a XSectionList object containing the cross sections 

369 :param slhafile: target file for writing the cross sections in SLHA format 

370 :param comment: optional comment to be added to each cross section block 

371 :param complain: complain if there are already cross sections in file 

372 

373 """ 

374 

375 if not os.path.isfile(slhafile): 

376 logger.error("SLHA file not found.") 

377 raise SModelSError() 

378 if len(xsecs) == 0: 

379 self.countNoXSecs+=1 

380 if self.countNoXSecs < 3: 

381 logger.warning("No cross sections available.") 

382 if self.countNoXSecs == 3: 

383 logger.warning("No cross sections available (will quench such warnings in future).") 

384 return False 

385 # Check if file already contain cross section blocks 

386 xSectionList = crossSection.getXsecFromSLHAFile(slhafile) 

387 if xSectionList and complain: 

388 logger.info("SLHA file already contains XSECTION blocks. Adding " 

389 "only missing cross sections.") 

390 

391 # Write cross sections to file, if they do not overlap any cross section in 

392 # the file 

393 outfile = open(slhafile, 'a') 

394 nxsecs = 0 

395 for xsec in xsecs: 

396 writeXsec = True 

397 for oldxsec in xSectionList: 

398 if oldxsec.info == xsec.info and set(oldxsec.pid) == set(xsec.pid): 

399 writeXsec = False 

400 break 

401 if writeXsec: 

402 nxsecs += 1 

403 outfile.write( self.xsecToBlock(xsec, (2212, 2212), comment) + "\n") 

404 outfile.close() 

405 

406 return nxsecs

408 def xsecToBlock( self, xsec, inPDGs=(2212, 2212), comment=None, xsecUnit = pb): 

409 """ 

410 Generate a string for a XSECTION block in the SLHA format from a XSection 

411 object. 

412 

413 :param inPDGs: defines the PDGs of the incoming states 

414 (default = 2212,2212) 

415 

416 :param comment: is added at the end of the header as a comment 

417 :param xsecUnit: unit of cross sections to be written (default is pb). 

418 Must be a Unum unit. 

419 

420 """ 

421 if type(xsec) != type(crossSection.XSection()): 

422 logger.error("Wrong input") 

423 raise SModelSError() 

424 # Sqrt(s) in GeV 

425 header = "XSECTION " + str(xsec.info.sqrts / GeV) 

426 for pdg in inPDGs: 

427 # PDGs of incoming states 

428 header += " " + str(pdg) 

429 # Number of outgoing states 

430 header += " " + str(len(xsec.pid)) 

431 for pid in xsec.pid: 

432 # PDGs of outgoing states 

433 header += " " + str(pid) 

434 if comment: 

435 header += " # " + str(comment) # Comment 

436 entry = " 0 " + str(xsec.info.order) + " 0 0 0 0 " + \ 

437 str( "%16.8E" % (xsec.value / xsecUnit) ) + " SModelSv" + \ 

438 installation.version() 

439 

440 return "\n" + header + "\n" + entry

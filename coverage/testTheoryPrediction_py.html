41 def checkAnalysis(self,expresult,smstoplist): 

42 expID = expresult.globalInfo.id 

43 theorypredictions = theoryPredictionsFor(expresult, smstoplist) 

44 defpreds=self.predictions() 

45 if not theorypredictions: 

46 print ( "no theory predictions for",expresult,"??" ) 

47 sys.exit(-1) 

48 for pred in theorypredictions: 

49 predval=pred.xsection.value 

50 defpredval = defpreds[expID] 

51 diff = abs ( predval.asNumber(fb) - defpredval.asNumber(fb) ) / defpredval.asNumber(fb) 

52 self.assertTrue ( diff < .1 ) 

53 #self.assertAlmostEqual( predval.asNumber(fb), defpredval.asNumber(fb), places=4 ) 

54 pred.computeStatistics( marginalize=True, deltas_rel=0. ) 

55 if pred.chi2 != self.predchi2()[expID]: 

56 diff = abs ( pred.chi2 - self.predchi2()[expID] ) / self.predchi2()[expID]

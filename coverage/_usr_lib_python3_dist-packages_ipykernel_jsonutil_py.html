49def encode_images(format_dict): 

50 """b64-encodes images in a displaypub format dict 

51 

52 Perhaps this should be handled in json_clean itself? 

53 

54 Parameters 

55 ---------- 

56 

57 format_dict : dict 

58 A dictionary of display data keyed by mime-type 

59 

60 Returns 

61 ------- 

62 

63 format_dict : dict 

64 A copy of the same dictionary, 

65 but binary image data ('image/png', 'image/jpeg' or 'application/pdf') 

66 is base64-encoded. 

67 

68 """ 

69 

70 # no need for handling of ambiguous bytestrings on Python 3, 

71 # where bytes objects always represent binary data and thus 

72 # base64-encoded. 

73 if py3compat.PY3: 

74 return format_dict

76 encoded = format_dict.copy() 

77 

78 pngdata = format_dict.get('image/png') 

79 if isinstance(pngdata, bytes): 

80 # make sure we don't double-encode 

81 if not pngdata.startswith(PNG64): 

82 pngdata = b2a_base64(pngdata) 

83 encoded['image/png'] = pngdata.decode('ascii') 

84 

85 jpegdata = format_dict.get('image/jpeg') 

86 if isinstance(jpegdata, bytes): 

87 # make sure we don't double-encode 

88 if not jpegdata.startswith(JPEG64): 

89 jpegdata = b2a_base64(jpegdata) 

90 encoded['image/jpeg'] = jpegdata.decode('ascii') 

91 

92 gifdata = format_dict.get('image/gif') 

93 if isinstance(gifdata, bytes): 

94 # make sure we don't double-encode 

95 if not gifdata.startswith((GIF_64, GIF89_64)): 

96 gifdata = b2a_base64(gifdata)

99 pdfdata = format_dict.get('application/pdf') 

100 if isinstance(pdfdata, bytes): 

101 # make sure we don't double-encode 

102 if not pdfdata.startswith(PDF64): 

103 pdfdata = b2a_base64(pdfdata) 

104 encoded['application/pdf'] = pdfdata.decode('ascii')

109def json_clean(obj): 

110 """Clean an object to ensure it's safe to encode in JSON. 

111 

112 Atomic, immutable objects are returned unmodified. Sets and tuples are 

113 converted to lists, lists are copied and dicts are also copied. 

114 

115 Note: dicts whose keys could cause collisions upon encoding (such as a dict 

116 with both the number 1 and the string '1' as keys) will cause a ValueError 

117 to be raised. 

118 

119 Parameters 

120 ---------- 

121 obj : any python object 

122 

123 Returns 

124 ------- 

125 out : object 

126 

127 A version of the input which will not cause an encoding error when 

128 encoded as JSON. Note that this function does not *encode* its inputs, 

129 it simply sanitizes it so that there will be no encoding errors later. 

130 

131 """ 

132 # types that are 'atomic' and ok in json as-is. 

133 atomic_ok = (unicode_type, type(None)) 

134 

135 # containers that we need to convert into lists 

136 container_to_list = (tuple, set, types.GeneratorType) 

137 

138 # Since bools are a subtype of Integrals, which are a subtype of Reals, 

139 # we have to check them in that order. 

140 

141 if isinstance(obj, bool): 

142 return obj 

143 

144 if isinstance(obj, numbers.Integral): 

145 # cast int to int, in case subclasses override __str__ (e.g. boost enum, #4598) 

146 return int(obj) 

147 

148 if isinstance(obj, numbers.Real): 

149 # cast out-of-range floats to their reprs 

150 if math.isnan(obj) or math.isinf(obj): 

151 return repr(obj) 

152 return float(obj) 

153 

154 if isinstance(obj, atomic_ok): 

155 return obj 

156 

157 if isinstance(obj, bytes): 

158 if py3compat.PY3: 

159 # unanmbiguous binary data is base64-encoded 

160 # (this probably should have happened upstream) 

161 return b2a_base64(obj).decode('ascii') 

162 else: 

163 # Python 2 bytestr is ambiguous, 

164 # needs special handling for possible binary bytestrings. 

165 # imperfect workaround: if ascii, assume text. 

166 # otherwise assume binary, base64-encode (py3 behavior). 

167 try: 

168 return obj.decode('ascii') 

169 except UnicodeDecodeError: 

170 return b2a_base64(obj).decode('ascii') 

171 

172 if isinstance(obj, container_to_list) or ( 

173 hasattr(obj, '__iter__') and hasattr(obj, next_attr_name)): 

174 obj = list(obj) 

175 

176 if isinstance(obj, list): 

177 return [json_clean(x) for x in obj] 

178 

179 if isinstance(obj, dict): 

180 # First, validate that the dict won't lose data in conversion due to 

181 # key collisions after stringification. This can happen with keys like 

182 # True and 'true' or 1 and '1', which collide in JSON. 

183 nkeys = len(obj) 

184 nkeys_collapsed = len(set(map(unicode_type, obj))) 

185 if nkeys != nkeys_collapsed: 

186 raise ValueError('dict cannot be safely converted to JSON: ' 

187 'key collision would lead to dropped values') 

188 # If all OK, proceed by making the new dict that will be json-safe 

189 out = {} 

190 for k,v in iteritems(obj): 

191 out[unicode_type(k)] = json_clean(v) 

192 return out 

193 if isinstance(obj, datetime): 

194 return obj.strftime(ISO8601) 

195 

196 # we don't understand it, it's probably an unserializable object 

197 raise ValueError("Can't clean for JSON: %r" % obj)

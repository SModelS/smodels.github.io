238class _BoundedIntRange(_IntRange):

243 min, max = kwargs.get('min', 0), kwargs.get('max', 100) 

244 if not kwargs.get('value', None): 

245 kwargs['value'] = (0.75 * min + 0.25 * max, 

246 0.25 * min + 0.75 * max) 

247 super(_BoundedIntRange, self).__init__(*args, **kwargs)

249 @validate('min', 'max') 

250 def _validate_bounds(self, proposal): 

251 trait = proposal['trait'] 

252 new = proposal['value'] 

253 if trait.name == 'min' and new > self.max: 

254 raise TraitError('setting min > max') 

255 if trait.name == 'max' and new < self.min: 

256 raise TraitError('setting max < min') 

257 if trait.name == 'min': 

258 self.value = (max(new, self.value[0]), max(new, self.value[1])) 

259 if trait.name == 'max': 

260 self.value = (min(new, self.value[0]), min(new, self.value[1])) 

261 return new 

262 

263 @validate('value') 

264 def _validate_value(self, proposal): 

265 lower, upper = super(_BoundedIntRange, self)._validate_value(proposal) 

266 lower, upper = min(lower, self.max), min(upper, self.max) 

267 lower, upper = max(lower, self.min), max(upper, self.min) 

268 return lower, upper

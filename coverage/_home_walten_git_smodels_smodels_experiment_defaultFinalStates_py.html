32numu = Particle(Z2parity=1, label='numu', pdg=14, mass=0.*MeV, eCharge=0, colordim=1, spin=1./2, totalwidth = 0.*GeV)

43q = MultiParticle('q', [u,d,s,u.chargeConjugate(),d.chargeConjugate(),s.chargeConjugate()]) 

44c = Particle(Z2parity=1, label='c', pdg=4, mass=1.28*GeV, eCharge=(2./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV)

51###Group b and b* in a single particle: 

52b = MultiParticle('b',[b,b.chargeConjugate('b')]) 

53###(We want to be able to differentiate between t+ and t-, so we define both particles:

55tC = t.chargeConjugate()

59g = Particle(Z2parity=1, label='g', pdg=21, mass=0.*MeV, eCharge=0, colordim=8, spin=1, totalwidth = 0.*GeV) 

60photon = Particle(Z2parity=1, label='photon',pdg=22, mass=0.*MeV, eCharge=0, colordim=1, spin=1, totalwidth = 0.*GeV, _isInvisible=False)

126#Define list of inclusive final states: 

127SMfinalStates = [eList,muList,taList,lpList,lmList,lList,WList,

134allFinalStates = SMfinalStates + BSMfinalStates 

135#Avoid double counting: 

136for i,ptc in enumerate(allFinalStates): 

137 if any((ptc is p and i != j) for j,p in enumerate(allFinalStates)): 

138 allFinalStates.remove(ptc)

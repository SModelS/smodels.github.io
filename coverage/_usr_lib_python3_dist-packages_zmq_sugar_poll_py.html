16class Poller(object): 

17 """A stateful poll interface that mirrors Python's built-in poll.""" 

18 sockets = None 

19 _map = {} 

20 

21 def __init__(self): 

22 self.sockets = [] 

23 self._map = {} 

24 

25 def __contains__(self, socket): 

26 return socket in self._map

28 def register(self, socket, flags=POLLIN|POLLOUT): 

29 """p.register(socket, flags=POLLIN|POLLOUT) 

30 

31 Register a 0MQ socket or native fd for I/O monitoring. 

32  

33 register(s,0) is equivalent to unregister(s). 

34 

35 Parameters 

36 ---------- 

37 socket : zmq.Socket or native socket 

38 A zmq.Socket or any Python object having a ``fileno()``  

39 method that returns a valid file descriptor. 

40 flags : int 

41 The events to watch for. Can be POLLIN, POLLOUT or POLLIN|POLLOUT. 

42 If `flags=0`, socket will be unregistered. 

43 """ 

44 if flags: 

45 if socket in self._map: 

46 idx = self._map[socket] 

47 self.sockets[idx] = (socket, flags) 

48 else: 

49 idx = len(self.sockets) 

50 self.sockets.append((socket, flags)) 

51 self._map[socket] = idx 

52 elif socket in self._map: 

53 # uregister sockets registered with no events 

54 self.unregister(socket) 

55 else: 

56 # ignore new sockets with no events 

57 pass 

58 

59 def modify(self, socket, flags=POLLIN|POLLOUT): 

60 """Modify the flags for an already registered 0MQ socket or native fd.""" 

61 self.register(socket, flags) 

62 

63 def unregister(self, socket): 

64 """Remove a 0MQ socket or native fd for I/O monitoring. 

65 

66 Parameters 

67 ---------- 

68 socket : Socket 

69 The socket instance to stop polling. 

70 """ 

71 idx = self._map.pop(socket) 

72 self.sockets.pop(idx) 

73 # shift indices after deletion 

74 for socket, flags in self.sockets[idx:]: 

75 self._map[socket] -= 1

77 def poll(self, timeout=None): 

78 """Poll the registered 0MQ or native fds for I/O. 

79 

80 Parameters 

81 ---------- 

82 timeout : float, int 

83 The timeout in milliseconds. If None, no `timeout` (infinite). This 

84 is in milliseconds to be compatible with ``select.poll()``. 

85 

86 Returns 

87 ------- 

88 events : list of tuples 

89 The list of events that are ready to be processed. 

90 This is a list of tuples of the form ``(socket, event)``, where the 0MQ Socket 

91 or integer fd is the first element, and the poll event mask (POLLIN, POLLOUT) is the second. 

92 It is common to call ``events = dict(poller.poll())``, 

93 which turns the list of tuples into a mapping of ``socket : event``. 

94 """ 

95 if timeout is None or timeout < 0: 

96 timeout = -1 

97 elif isinstance(timeout, float): 

98 timeout = int(timeout) 

99 return zmq_poll(self.sockets, timeout=timeout)

102def select(rlist, wlist, xlist, timeout=None): 

103 """select(rlist, wlist, xlist, timeout=None) -> (rlist, wlist, xlist) 

104 

105 Return the result of poll as a lists of sockets ready for r/w/exception. 

106 

107 This has the same interface as Python's built-in ``select.select()`` function. 

108 

109 Parameters 

110 ---------- 

111 timeout : float, int, optional 

112 The timeout in seconds. If None, no timeout (infinite). This is in seconds to be 

113 compatible with ``select.select()``. 

114 rlist : list of sockets/FDs 

115 sockets/FDs to be polled for read events 

116 wlist : list of sockets/FDs 

117 sockets/FDs to be polled for write events 

118 xlist : list of sockets/FDs 

119 sockets/FDs to be polled for error events 

120  

121 Returns 

122 ------- 

123 (rlist, wlist, xlist) : tuple of lists of sockets (length 3) 

124 Lists correspond to sockets available for read/write/error events respectively. 

125 """ 

126 if timeout is None: 

127 timeout = -1 

128 # Convert from sec -> us for zmq_poll. 

129 # zmq_poll accepts 3.x style timeout in ms 

130 timeout = int(timeout*1000.0) 

131 if timeout < 0: 

132 timeout = -1 

133 sockets = [] 

134 for s in set(rlist + wlist + xlist): 

135 flags = 0 

136 if s in rlist: 

137 flags |= POLLIN 

138 if s in wlist: 

139 flags |= POLLOUT 

140 if s in xlist: 

141 flags |= POLLERR 

142 sockets.append((s, flags)) 

143 return_sockets = zmq_poll(sockets, timeout) 

144 rlist, wlist, xlist = [], [], [] 

145 for s, flags in return_sockets: 

146 if flags & POLLIN: 

147 rlist.append(s) 

148 if flags & POLLOUT: 

149 wlist.append(s) 

150 if flags & POLLERR: 

151 xlist.append(s) 

152 return rlist, wlist, xlist

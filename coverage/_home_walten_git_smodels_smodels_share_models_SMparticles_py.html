1""" 

2.. module:: SMparticleDefinitions 

3 :synopsis: Defines the SM particles.

9 

10from smodels.tools.physicsUnits import MeV , GeV 

11from smodels.theory.particle import Particle, MultiParticle 

12 

13# SM particles 

14 

15##Charged leptons: 

16e = Particle(Z2parity=1, label='e-', pdg=11, mass=0.5*MeV, eCharge=-1, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

17mu = Particle(Z2parity=1, label='mu-', pdg=13, mass=106.*MeV, eCharge=-1, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

18ta = Particle(Z2parity=1, label='ta-', pdg=15, mass=1777.*MeV, eCharge=-1, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

19eC = e.chargeConjugate() 

20muC = mu.chargeConjugate() 

21taC = ta.chargeConjugate() 

22##----------------------------------------------------------------------------------------------------------------------------- 

23##Neutrinos 

24nue = Particle(Z2parity=1, label='nue', pdg=12, mass=0.*MeV, eCharge=0, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

25numu = Particle(Z2parity=1, label='numu', pdg=14, mass=0.*MeV, eCharge=0, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

26nuta = Particle(Z2parity=1, label='nuta', pdg=16, mass=0.*MeV, eCharge=0, colordim=1, spin=1./2, totalwidth = 0.*GeV) 

27###Group all neutrinos into a single particle: 

28nu = MultiParticle('nu',[nue,numu,nuta,nue.chargeConjugate(),numu.chargeConjugate(),nuta.chargeConjugate()]) 

29##----------------------------------------------------------------------------------------------------------------------------- 

30 

31##Light quarks: 

32d = Particle(Z2parity=1, label='d', pdg=1, mass=0.*MeV, eCharge=(-1./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV) 

33u = Particle(Z2parity=1, label='u', pdg=2, mass=0.*MeV, eCharge=(2./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV) 

34s = Particle(Z2parity=1, label='s', pdg=3, mass=0.*MeV, eCharge=(-1./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV) 

35#Group all light quarks in a single particle 

36q = MultiParticle('q', [u,d,s,u.chargeConjugate(),d.chargeConjugate(),s.chargeConjugate()]) 

37c = Particle(Z2parity=1, label='c', pdg=4, mass=1.28*GeV, eCharge=(2./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV) 

38###Group c and c* in a single particle: 

39c = MultiParticle('c',[c,c.chargeConjugate('c')]) 

40##----------------------------------------------------------------------------------------------------------------------------- 

41 

42##Heavy quarks: 

43b = Particle(Z2parity=1, label='b', pdg=5, mass=4.7*GeV, eCharge=(-1./3.), colordim=3, spin=1./2, totalwidth = 0.*GeV) 

44###Group b and b* in a single particle: 

45b = MultiParticle('b',[b,b.chargeConjugate('b')]) 

46###(We want to be able to differentiate between t+ and t-, so we define both particles: 

47t = Particle(Z2parity=1, label='t+', pdg=6, mass=173.1*GeV, eCharge=(2./3.), colordim=3, spin=1./2, totalwidth=1.4*GeV) 

48tC = t.chargeConjugate() 

49##----------------------------------------------------------------------------------------------------------------------------- 

50 

51##Gauge bosons: 

52g = Particle(Z2parity=1, label='g', pdg=21, mass=0.*MeV, eCharge=0, colordim=8, spin=1, totalwidth = 0.*GeV) 

53photon = Particle(Z2parity=1, label='photon',pdg=22, mass=0.*MeV, eCharge=0, colordim=1, spin=1, totalwidth = 0.*GeV, _isInvisible=False) 

54Z = Particle(Z2parity=1, label='Z', pdg=23, mass=91.*GeV , eCharge=0, colordim=1, spin=1, totalwidth=2.5*GeV, _isInvisible=False) 

55#We group each electrically neutral gauge boson with its antiparticle: 

56g = MultiParticle('g',[g,g.chargeConjugate('g')]) 

57photon = MultiParticle('photon',[photon,photon.chargeConjugate('photon')]) 

58Z = MultiParticle('Z',[Z,Z.chargeConjugate('Z')]) 

59W = Particle(Z2parity=1, label='W+', pdg=24, mass=80.*GeV, eCharge=1, colordim=1, spin=1, totalwidth=2.0*GeV) 

60WC = W.chargeConjugate() 

61##----------------------------------------------------------------------------------------------------------------------------- 

62 

63##Higgs: 

64higgs = Particle(Z2parity=1, label='higgs', pdg=25, mass=125.*GeV, eCharge=0, colordim=1, spin=0, totalwidth = 0.*GeV, _isInvisible=False) 

65###Group higgs and conjugate in single particle: 

66higgs = MultiParticle('higgs',[higgs,higgs.chargeConjugate('higgs')]) 

67##----------------------------------------------------------------------------------------------------------------------------- 

68 

69##Pions 

70pip = Particle(Z2parity=1, label='pi+', pdg=211, mass=140.*MeV, eCharge=+1, colordim=1, spin=0, totalwidth = 0.*GeV) 

71piz = Particle(Z2parity=1, label='pi0', pdg=111, mass=140.*MeV, eCharge=0, colordim=1, spin=0, totalwidth = 0.*GeV) 

72###Group all pions in a single particle: 

73pion = MultiParticle('pion',[pip,piz,pip.chargeConjugate(),piz.chargeConjugate('pi0')]) 

74 

75leptons = [e,mu,ta,eC,muC,taC,nu] 

76gauge = [g,photon,Z,W,WC] 

77quarks = [q,c,b,t,tC] 

78 

79SMList = leptons + gauge + quarks + [higgs,pion] 

80SMparticleList = MultiParticle('SM',SMList)

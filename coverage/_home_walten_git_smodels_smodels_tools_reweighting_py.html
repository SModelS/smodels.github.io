15def defaultEffReweight(element,minWeight=1e-10): 

16 """ 

17 Computes the lifetime reweighting factor for the element efficiency 

18 based on the lifetimes of all intermediate particles and the last stable odd-particle 

19 appearing in the element. 

20 The fraction corresponds to the fraction of decays corresponding to prompt 

21 decays to all intermediate BSM particles and to a long-lived decay (outside the detector) 

22 to the final BSM state. 

23 

24 :param element: Element object or nested list of widths 

25 :param minWeight: Lower cut for the reweighting factor. Any value below this will be taken to be zero. 

26 

27 :return: Reweight factor (float) 

28 """ 

29 

30 if isinstance(element,list): 

31 elWidths = [[w.asNumber(GeV) for w in br] for br in element] 

32 else: 

33 elWidths = [ [ x.asNumber(GeV) for x in L ] for L in element.totalwidth ] 

34 elFraction = 1. 

35 for branchWidths in elWidths: 

36 branchFraction = 1. 

37 for width in branchWidths[:-1]: 

38 branchFraction *= calculateProbabilities(width)['F_prompt'] 

39 branchFraction *= calculateProbabilities(width=branchWidths[-1])['F_long'] 

40 elFraction *= branchFraction 

41 

42 if elFraction < minWeight: 

43 return 0. 

44 

45 return elFraction

47def defaultULReweight(element): 

48 """ 

49 Computes the lifetime reweighting factor for the element upper limit 

50 based on the lifetimes of all intermediate particles and the last stable odd-particle 

51 appearing in the element. 

52 The fraction corresponds to the fraction of decays corresponding to prompt 

53 decays to all intermediate BSM particles and to a long-lived decay (outside the detector) 

54 to the final BSM state. 

55 

56 :param element: Element object 

57 

58 :return: Reweight factor (float) 

59 """ 

60 

61 effFactor = defaultEffReweight(element) 

62 if not effFactor: 

63 return None 

64 else: 

65 return 1./effFactor

68def reweightFactorFor(element,resType='prompt'): 

69 """ 

70 Computer the reweighting factor for the element according to the experimental result type. 

71 Currently only two result types are supported: 'prompt' and 'displaced'. 

72 If resultType = 'prompt', returns the reweighting factor for all decays in the element 

73 to be prompt and the last odd particle to be stable. 

74 If resultType = 'displaced', returns the reweighting factor for ANY decay in the element 

75 to be displaced and no long-lived decays and the last odd particle to be stable. 

76 Not that the fraction of "long-lived (meta-stable) decays" is usually included 

77 in topologies where the meta-stable particle appears in the final state. Hence 

78 it should not be included in the prompt or displaced fractions. 

79 

80 :param element: Element object 

81 :param resType: Type of result to compute the reweight factor for (either 'prompt' or 'displaced') 

82 :return: probabilities (depending on types of decay within branch), branches (with different labels depending on type of decay) 

83 """ 

84 

85 if not isinstance(element,Element): 

86 logger.error('element should be an Element object and not %s' %type(element)) 

87 raise SModelSError() 

88 rType = resType.lower() 

89 if not rType in ['prompt', 'displaced']: 

90 logger.error('resultType should be prompt or displaced and not %s' %str(resType)) 

91 raise SModelSError() 

92 

93 FpromptTotal = 1. #Keep track of the probability of all decays being prompt 

94 FlongTotal = 1. #Keep track of the probability for at least one decay being long-lived 

95 for branch in element.branches: 

96 for particle in branch.oddParticles[:-1]: 

97 if isinstance(particle.totalwidth, list): 

98 width = min(particle.totalwidth) 

99 else: 

100 width = particle.totalwidth 

101 if width == float('inf')*GeV: 

102 Fprompt = 1. 

103 Fdisp = 0. 

104 elif width == 0.*GeV: 

105 Fprompt = 0. 

106 Fdisp = 0. 

107 else: 

108 probs = calculateProbabilities(width.asNumber(GeV)) 

109 Fprompt = probs['F_prompt'] 

110 Fdisp = probs['F_displaced'] 

111 FlongTotal *= (Fprompt+Fdisp) 

112 FpromptTotal *= Fprompt 

113 #Prob(at least one decay being long-lived) = 1 - Prob(all decays being either prompt or displaced) 

114 FlongTotal = 1. - FlongTotal 

115 #Prob(at least one displaced decay and no long-lived decays) = 1 -Prob(at least one long-lived) - Prob(all prompt) 

116 FdispTotal = 1. - FlongTotal - FpromptTotal 

117 #Include Flong factor for final BSM states: 

118 for branch in element.branches: 

119 width = branch.oddParticles[-1].totalwidth 

120 if width == float('inf')*GeV: 

121 Flong = 0. 

122 elif width == 0*GeV: 

123 Flong = 1. 

124 else: 

125 Flong = calculateProbabilities(width.asNumber(GeV))['F_long'] 

126 FdispTotal *= Flong 

127 FpromptTotal *= Flong 

128 

129 if rType == 'prompt': 

130 return FpromptTotal 

131 if rType == 'displaced': 

132 return FdispTotal

134def calculateProbabilities(width, l_inner = .001, 

135 gb_inner=1.3,l_outer=10.,gb_outer=1.43): 

136 """ 

137 The fraction of prompt and displaced decays are defined as: 

138  

139 F_long = exp(-totalwidth*l_outer/gb_outer) 

140 F_prompt = 1 - exp(-totaltotalwidth*l_inner/gb_inner) 

141 F_displaced = 1 - F_prompt - F_long 

142  

143 :param l_inner: is the inner radius of the detector, given in meters 

144 :param l_outer: is the outer radius of the detector, given in meters 

145 :param gb_inner: is the estimate for the kinematical factor gamma*beta for prompt decays. 

146 :param gb_outer: is the estimate for the kinematical factor gamma*beta for decays outside the detector.  

147 :param width: particle width for which probabilities should be calculated (in GeV) 

148  

149 :return: Dictionary with the probabilities for the particle not to decay (in the detector), to decay promptly or displaced. 

150 """ 

151 

152 # hc = 197.327*GeV*fm #hbar * c 

153 hc = 197.327*1e-18 # hbar * c * m / GeV, expecting the width to be in GeV, and lengths in m 

154 

155 F_long = exp(-width*l_outer/(gb_outer*hc)) 

156 F_prompt = 1. - exp(-width*l_inner/(gb_inner*hc)) 

157 F_displaced = 1. - F_prompt - F_long 

158 

159 return {'F_long' : F_long, 'F_prompt' : F_prompt, 'F_displaced' : F_displaced}

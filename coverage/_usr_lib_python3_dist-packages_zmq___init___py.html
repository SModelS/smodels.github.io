10 dlopen = hasattr(sys, 'getdlopenflags') # unix-only 

11 # RTLD flags are added to os in Python 3 

12 # get values from os because ctypes values are WRONG on pypy 

13 PYPY = platform.python_implementation().lower() == 'pypy' 

14 

15 if dlopen: 

16 dlflags = sys.getdlopenflags() 

17 # set RTLD_GLOBAL, unset RTLD_LOCAL 

18 flags = ctypes.RTLD_GLOBAL | dlflags 

19 # ctypes.RTLD_LOCAL is 0 on pypy, which is *wrong* 

20 flags &= ~ getattr(os, 'RTLD_LOCAL', 4) 

21 # pypy on darwin needs RTLD_LAZY for some reason 

22 if PYPY and sys.platform == 'darwin': 

23 flags |= getattr(os, 'RTLD_LAZY', 1) 

24 flags &= ~ getattr(os, 'RTLD_NOW', 2) 

25 sys.setdlopenflags(flags)

29 pass

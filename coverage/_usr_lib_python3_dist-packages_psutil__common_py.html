267def usage_percent(used, total, round_=None): 

268 """Calculate percentage usage of 'used' against 'total'.""" 

269 try: 

270 ret = (used / total) * 100 

271 except ZeroDivisionError: 

272 ret = 0.0 if isinstance(used, float) or isinstance(total, float) else 0 

273 if round_ is not None: 

274 return round(ret, round_) 

275 else: 

276 return ret

279def memoize(fun): 

280 """A simple memoize decorator for functions supporting (hashable) 

281 positional arguments. 

282 It also provides a cache_clear() function for clearing the cache: 

283 

284 >>> @memoize 

285 ... def foo() 

286 ... return 1 

287 ... 

288 >>> foo() 

289 1 

290 >>> foo.cache_clear() 

291 >>> 

292 """ 

293 @functools.wraps(fun) 

294 def wrapper(*args, **kwargs): 

295 key = (args, frozenset(sorted(kwargs.items()))) 

296 try: 

297 return cache[key] 

298 except KeyError: 

299 ret = cache[key] = fun(*args, **kwargs) 

300 return ret 

301 

302 def cache_clear(): 

303 """Clear cache.""" 

304 cache.clear() 

305 

306 cache = {} 

307 wrapper.cache_clear = cache_clear 

308 return wrapper

311def memoize_when_activated(fun): 

312 """A memoize decorator which is disabled by default. It can be 

313 activated and deactivated on request. 

314 For efficiency reasons it can be used only against class methods 

315 accepting no arguments. 

316 

317 >>> class Foo: 

318 ... @memoize 

319 ... def foo() 

320 ... print(1) 

321 ... 

322 >>> f = Foo() 

323 >>> # deactivated (default) 

324 >>> foo() 

325 1 

326 >>> foo() 

327 1 

328 >>> 

329 >>> # activated 

330 >>> foo.cache_activate(self) 

331 >>> foo() 

332 1 

333 >>> foo() 

334 >>> foo() 

335 >>> 

336 """ 

337 @functools.wraps(fun) 

338 def wrapper(self): 

339 try: 

340 # case 1: we previously entered oneshot() ctx 

341 ret = self._cache[fun] 

342 except AttributeError: 

343 # case 2: we never entered oneshot() ctx 

344 return fun(self) 

345 except KeyError: 

346 # case 3: we entered oneshot() ctx but there's no cache 

347 # for this entry yet 

348 ret = self._cache[fun] = fun(self) 

349 return ret

351 def cache_activate(proc): 

352 """Activate cache. Expects a Process instance. Cache will be 

353 stored as a "_cache" instance attribute.""" 

354 proc._cache = {} 

355 

356 def cache_deactivate(proc): 

357 """Deactivate and clear cache.""" 

358 try: 

359 del proc._cache 

360 except AttributeError: 

361 pass 

362 

363 wrapper.cache_activate = cache_activate 

364 wrapper.cache_deactivate = cache_deactivate 

365 return wrapper

368def isfile_strict(path): 

369 """Same as os.path.isfile() but does not swallow EACCES / EPERM 

370 exceptions, see: 

371 http://mail.python.org/pipermail/python-dev/2012-June/120787.html 

372 """ 

373 try: 

374 st = os.stat(path) 

375 except OSError as err: 

376 if err.errno in (errno.EPERM, errno.EACCES): 

377 raise 

378 return False 

379 else: 

380 return stat.S_ISREG(st.st_mode)

412def parse_environ_block(data): 

413 """Parse a C environ block of environment variables into a dictionary.""" 

414 # The block is usually raw data from the target process. It might contain 

415 # trailing garbage and lines that do not look like assignments. 

416 ret = {} 

417 pos = 0 

418 

419 # localize global variable to speed up access. 

420 WINDOWS_ = WINDOWS 

421 while True: 

422 next_pos = data.find("\0", pos) 

423 # nul byte at the beginning or double nul byte means finish 

424 if next_pos <= pos: 

425 break 

426 # there might not be an equals sign 

427 equal_pos = data.find("=", pos, next_pos) 

428 if equal_pos > pos: 

429 key = data[pos:equal_pos] 

430 value = data[equal_pos + 1:next_pos] 

431 # Windows expects environment variables to be uppercase only 

432 if WINDOWS_: 

433 key = key.upper() 

434 ret[key] = value 

435 pos = next_pos + 1 

436 

437 return ret

466def deprecated_method(replacement): 

467 """A decorator which can be used to mark a method as deprecated 

468 'replcement' is the method name which will be called instead. 

469 """ 

470 def outer(fun): 

471 msg = "%s() is deprecated and will be removed; use %s() instead" % ( 

472 fun.__name__, replacement) 

473 if fun.__doc__ is None: 

474 fun.__doc__ = msg 

475 

476 @functools.wraps(fun) 

477 def inner(self, *args, **kwargs): 

478 warnings.warn(msg, category=FutureWarning, stacklevel=2) 

479 return getattr(self, replacement)(*args, **kwargs) 

480 return inner 

481 return outer

484class _WrapNumbers: 

485 """Watches numbers so that they don't overflow and wrap 

486 (reset to zero). 

487 """ 

488 

489 def __init__(self): 

490 self.lock = threading.Lock() 

491 self.cache = {} 

492 self.reminders = {} 

493 self.reminder_keys = {} 

494 

495 def _add_dict(self, input_dict, name): 

496 assert name not in self.cache 

497 assert name not in self.reminders 

498 assert name not in self.reminder_keys 

499 self.cache[name] = input_dict 

500 self.reminders[name] = defaultdict(int) 

501 self.reminder_keys[name] = defaultdict(set) 

502 

503 def _remove_dead_reminders(self, input_dict, name): 

504 """In case the number of keys changed between calls (e.g. a 

505 disk disappears) this removes the entry from self.reminders. 

506 """ 

507 old_dict = self.cache[name] 

508 gone_keys = set(old_dict.keys()) - set(input_dict.keys()) 

509 for gone_key in gone_keys: 

510 for remkey in self.reminder_keys[name][gone_key]: 

511 del self.reminders[name][remkey] 

512 del self.reminder_keys[name][gone_key] 

513 

514 def run(self, input_dict, name): 

515 """Cache dict and sum numbers which overflow and wrap. 

516 Return an updated copy of `input_dict` 

517 """ 

518 if name not in self.cache: 

519 # This was the first call. 

520 self._add_dict(input_dict, name) 

521 return input_dict 

522 

523 self._remove_dead_reminders(input_dict, name) 

524 

525 old_dict = self.cache[name] 

526 new_dict = {} 

527 for key in input_dict.keys(): 

528 input_tuple = input_dict[key] 

529 try: 

530 old_tuple = old_dict[key] 

531 except KeyError: 

532 # The input dict has a new key (e.g. a new disk or NIC) 

533 # which didn't exist in the previous call. 

534 new_dict[key] = input_tuple 

535 continue 

536 

537 bits = [] 

538 for i in range(len(input_tuple)): 

539 input_value = input_tuple[i] 

540 old_value = old_tuple[i] 

541 remkey = (key, i) 

542 if input_value < old_value: 

543 # it wrapped! 

544 self.reminders[name][remkey] += old_value 

545 self.reminder_keys[name][key].add(remkey) 

546 bits.append(input_value + self.reminders[name][remkey]) 

547 

548 new_dict[key] = tuple(bits) 

549 

550 self.cache[name] = input_dict 

551 return new_dict 

552 

553 def cache_clear(self, name=None): 

554 """Clear the internal cache, optionally only for function 'name'.""" 

555 with self.lock: 

556 if name is None: 

557 self.cache.clear() 

558 self.reminders.clear() 

559 self.reminder_keys.clear() 

560 else: 

561 self.cache.pop(name, None) 

562 self.reminders.pop(name, None) 

563 self.reminder_keys.pop(name, None) 

564 

565 def cache_info(self): 

566 """Return internal cache dicts as a tuple of 3 elements.""" 

567 with self.lock: 

568 return (self.cache, self.reminders, self.reminder_keys)

572 """Given an `input_dict` and a function `name`, adjust the numbers

575 """
